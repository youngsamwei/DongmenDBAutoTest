int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_update 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成update操作
     * */
   // fprintf(stderr, "TODO: update is not implemented yet. in plan_execute_update \n");


    //记录被更新条数
    size_t  updatedcountn=0;
    //用physical_scan_generate创建一个物理计划
    physical_scan *scan = physical_scan_generate(db, sqlStmtUpdate->where, tx);
    scan->beforeFirst(scan);
    //用循环遍历plan
    while (scan->next(scan)){
        for (size_t i = 0; i < sqlStmtUpdate->fieldsExpr->size; i++) {
            //获得字段的名字
            char *CurrentFieldName = arraylist_get(sqlStmtUpdate->fields,i);
            //分配存储空间
            variant *val = (variant*)calloc(sizeof(variant),1);
            //获取数据类型
            enum data_type field_type=scan->getField(scan, sqlStmtUpdate->tableName, CurrentFieldName)->type;
            physical_scan_evaluate_expression(arraylist_get(sqlStmtUpdate->fieldsExpr,i),scan,val);

            //判断数据类型是否匹配6
            if(val->type != field_type) {
                //若不匹配，报错
                fprintf(stdout,"error：data type can't match.");
                //返回非法的sql语句错误,停止
                return DONGMENDB_EINVALIDSQL;
            }
            if(val->type == DATA_TYPE_INT){
                // 使用scan设置int值
                scan->setInt(scan, sqlStmtUpdate->tableName, CurrentFieldName, val->intValue);
                //若设置的长度超出了定义的字符串长度，要截取处理
            }else if (val->type == DATA_TYPE_CHAR){
                scan->setString(scan, sqlStmtUpdate->tableName, CurrentFieldName, val->strValue);
            }
        }
        //记录条数增加
       updatedcountn++;
    }
    //关闭scan
    scan->close(scan);
    return updatedcountn;
};

#include <parser/statement.h>

/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

/*TODO: parse_sql_stmt_update， update语句解析*/

sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {
//    fprintf(stderr, "TODO: update is not implemented yet. in parse_sql_stmt_update \n");

    char *TN= NULL;
    arraylist *Fields = arraylist_create();//set fields 被更新的字段列表。
    arraylist *FieldsExpr = arraylist_create();//set fields expression 新值(表达式)列表 。
    TokenT *token = parseNextToken(parser);
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {//匹配update
        strcpy(parser->parserMessage, "Error: update should be into.Please add it.");
        return NULL;
    }
    token = parseNextToken(parser);
    if (token->type == TOKEN_WORD) {//解析表名
        TN = new_id_name();
        strcpy(TN, token->text);
    } else {
        strcpy(parser->parserMessage, "Error: missing table name.Please add it.");
        return NULL;
    }
    char  a=".";

    token = parseEatAndNextToken(parser);
    if(!matchToken(parser,TOKEN_RESERVED_WORD,"set")){//匹配set
        strcpy(parser->parserMessage, "Error:  set should be into.Please add it.");
        return NULL;
    }

    token = parseNextToken(parser);

    while (token != NULL && (token->type == TOKEN_WORD || token->type == TOKEN_COMMA)){
        if (token->type == TOKEN_COMMA){
            /*set之间可能有很多表达式，表达式之间是逗号，判断是否是逗号，若是逗号，则需要跳过逗号*/
            token = parseEatAndNextToken(parser);
        }

        /*获取字段名*/
        if(token->type == TOKEN_WORD)
        {
            char *FName = new_id_name();
            strcpy(FName,token->text);
            arraylist_add(Fields,FName);
        }
        else{
            strcpy(parser->parserMessage,"Error:missing field name.Please add it.");
            return NULL;
        }
        token = parseEatAndNextToken(parser);
        /*字段名后面跟着等号*/
        if (!matchToken(parser, TOKEN_EQ, "=")) {
            strcpy(parser->parserMessage, "Error: missing the key's symbol = .Please add it.");
            return NULL;
        }
        /*获取等号后面的表达式*/
        Expression *Expr = parseExpressionRD(parser);
        arraylist_add(FieldsExpr, Expr);
        token=parseNextToken(parser);
        if(token==NULL)
            break;
    }

    Expression *WExpr = NULL;
    if (token !=NULL) {
        if (!matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
            strcpy(parser->parserMessage,"Error: should be where.Please add it.");
            return NULL;
        }
        WExpr = parseExpressionRD(parser);
        if (parser->parserStateType == PARSER_WRONG) {
            return NULL;
        }
    }

    sql_stmt_update *sqlStmtUpdate = (sql_stmt_update *) malloc(sizeof(sql_stmt_update));
    sqlStmtUpdate->tableName = TN;
    sqlStmtUpdate->fields = Fields;
    sqlStmtUpdate->fieldsExpr = FieldsExpr;

    TableReference_t *al =   TableReference_make(TN, NULL);
    SRA_t *Tal =  SRATable(al);
    sqlStmtUpdate->where = Tal;
    if (WExpr!= NULL) {
        sqlStmtUpdate->where = SRASelect(Tal, WExpr);
    }
    return sqlStmtUpdate;
};
