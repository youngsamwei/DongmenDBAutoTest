#include <parser/statement.h>

/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

/*TODO: parse_sql_stmt_update， update语句解析*/
sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {
//    fprintf(stderr, "TODO: update is not implemented yet. in parse_sql_stmt_update \n");
    
    char *TableNameYW = NULL;
    arraylist *FieldsYW = arraylist_create();
    arraylist *FieldsExprYW = arraylist_create();
    TokenT *tokenYW = parseNextToken(parser);
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {
        strcpy(parser->parserMessage, "invalid sql: missing update.");
        return NULL;
    }
    tokenYW = parseNextToken(parser);
    if (tokenYW->type == TOKEN_WORD) {
        TableNameYW = new_id_name();
        strcpy(TableNameYW, tokenYW->text);
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }

    tokenYW = parseEatAndNextToken(parser);
    if(!matchToken(parser,TOKEN_RESERVED_WORD,"set")){
        strcpy(parser->parserMessage, "invalid sql: should be set.");
        return NULL;
    }

    tokenYW = parseNextToken(parser);

    while (tokenYW != NULL && (tokenYW->type == TOKEN_WORD || tokenYW->type == TOKEN_COMMA)){
        if (tokenYW->type == TOKEN_COMMA){
            /*set之间可能有很多表达式，表达式之间是逗号，判断是否是逗号，若是逗号，则需要跳过逗号*/
            tokenYW = parseEatAndNextToken(parser);
        }

        /*获取字段名*/
        if(tokenYW->type == TOKEN_WORD)
        {
            char *FieldNameYW = new_id_name();
            strcpy(FieldNameYW,tokenYW->text);
            arraylist_add(FieldsYW,FieldNameYW);
        }
        else{
            strcpy(parser->parserMessage,"invalid sql:missing field name.");
            return NULL;
        }
        tokenYW = parseEatAndNextToken(parser);
        /*字段名后面跟着等号*/
        if (!matchToken(parser, TOKEN_EQ, "=")) {
            strcpy(parser->parserMessage, "invalid sql: missing =.");
            return NULL;
        }
        /*获取等号后面的表达式*/
        Expression *ExprYW = parseExpressionRD(parser);
        arraylist_add(FieldsExprYW, ExprYW);
        tokenYW=parseNextToken(parser);
        if(tokenYW==NULL)
            break;
    }

    Expression *WhereExprYW = NULL;
    if (tokenYW !=NULL) {
        if (!matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
            strcpy(parser->parserMessage,"invalid sql: should be where.");
            return NULL;
        }
        WhereExprYW = parseExpressionRD(parser);
        if (parser->parserStateType == PARSER_WRONG) {
            return NULL;
        }
    }

    sql_stmt_update *sqlStmtUpdate = (sql_stmt_update *) malloc(sizeof(sql_stmt_update));
    sqlStmtUpdate->tableName = TableNameYW;
    sqlStmtUpdate->fields = FieldsYW;
    sqlStmtUpdate->fieldsExpr = FieldsExprYW;

    TableReference_t *RefYW =   TableReference_make(TableNameYW, NULL);
    SRA_t *TableYW =  SRATable(RefYW);
    sqlStmtUpdate->where = TableYW;
    if (WhereExprYW != NULL) {
        sqlStmtUpdate->where = SRASelect(TableYW, WhereExprYW);
    }
    return sqlStmtUpdate;
};

int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_update 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成update操作
     * */
    fprintf(stderr, "TODO: update is not implemented yet. in plan_execute_update \n");


    //记录被更新条数
    size_t  Updated_CountYW=0;
    //用physical_scan_generate创建一个物理计划
    physical_scan *scanYW = physical_scan_generate(db, sqlStmtUpdate->where, tx);
    scanYW->beforeFirst(scanYW);
    //用循环遍历plan
    while (scanYW->next(scanYW)){
        for (size_t i = 0; i < sqlStmtUpdate->fieldsExpr->size; i++) {
            //获得字段的名字
            char *CurrentFieldNameYW = arraylist_get(sqlStmtUpdate->fields,i);
            //分配存储空间
            variant *varYW = (variant*)calloc(sizeof(variant),1);

            //获取数据类型
            enum data_type field_type=scanYW->getField(scanYW, sqlStmtUpdate->tableName, CurrentFieldNameYW)->type;
            physical_scan_evaluate_expression(arraylist_get(sqlStmtUpdate->fieldsExpr,i),scanYW,varYW);

            //判断数据类型是否匹配6
            if(varYW->type != field_type) {
                //若不匹配，报错
                fprintf(stdout,"error：数据类型不匹配");
                //返回非法的sql语句错误,停止
                return DONGMENDB_EINVALIDSQL;
            }
            if (varYW->type == DATA_TYPE_CHAR){
                scanYW->setString(scanYW, sqlStmtUpdate->tableName, CurrentFieldNameYW, varYW->strValue);
            }else if(varYW->type == DATA_TYPE_INT){
                // 使用scan设置int值
                scanYW->setInt(scanYW, sqlStmtUpdate->tableName, CurrentFieldNameYW, varYW->intValue);
                //若设置的长度超出了定义的字符串长度，要截取处理
            }
        }
        //记录条数增加
        Updated_CountYW++;
    }
    //关闭scan
    scanYW->close(scanYW);
    return Updated_CountYW;
};
