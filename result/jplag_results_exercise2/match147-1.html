#include <parser/statement.h>

/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

/*TODO: parse_sql_stmt_update， update语句解析*/

sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {
//    fprintf(stderr, "TODO: update is not implemented yet. in parse_sql_stmt_update \n");
    char *TableNameXXCCF = NULL;
    arraylist *FieldsXXCCF = arraylist_create();
    arraylist *FieldsExprXXCCF = arraylist_create();
    TokenT *tokenXXCCF = parseNextToken(parser);
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {
        strcpy(parser->parserMessage, "invalid sql: missing update.");
        return NULL;
    }
    tokenXXCCF = parseNextToken(parser);
    if (tokenXXCCF->type == TOKEN_WORD) {
        TableNameXXCCF = new_id_name();
        strcpy(TableNameXXCCF, tokenXXCCF->text);
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }

    tokenXXCCF = parseEatAndNextToken(parser);
    if(!matchToken(parser,TOKEN_RESERVED_WORD,"set")){
        strcpy(parser->parserMessage, "invalid sql: should be set.");
        return NULL;
    }

    tokenXXCCF = parseNextToken(parser);

    while (tokenXXCCF != NULL && (tokenXXCCF->type == TOKEN_WORD || tokenXXCCF->type == TOKEN_COMMA)){
        if (tokenXXCCF->type == TOKEN_COMMA){
            /*set之间可能有很多表达式，表达式之间是逗号，判断是否是逗号，若是逗号，则需要跳过逗号*/
            tokenXXCCF = parseEatAndNextToken(parser);
        }

        /*获取字段名*/
        if(tokenXXCCF->type == TOKEN_WORD)
        {
            char *FieldNameXXCCF = new_id_name();
            strcpy(FieldNameXXCCF,tokenXXCCF->text);
            arraylist_add(FieldsXXCCF,FieldNameXXCCF);
        }
        else{
            strcpy(parser->parserMessage,"invalid sql:missing field name.");
            return NULL;
        }
        tokenXXCCF = parseEatAndNextToken(parser);
        /*字段名后面跟着等号*/
        if (!matchToken(parser, TOKEN_EQ, "=")) {
            strcpy(parser->parserMessage, "invalid sql: missing =.");
            return NULL;
        }
        /*获取等号后面的表达式*/
        Expression *ExprXXCCF = parseExpressionRD(parser);
        arraylist_add(FieldsExprXXCCF, ExprXXCCF);
        tokenXXCCF=parseNextToken(parser);
        if(tokenXXCCF==NULL)
            break;
    }

    Expression *WhereExprXXCCF = NULL;
    if (tokenXXCCF !=NULL) {
        if (!matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
            strcpy(parser->parserMessage,"invalid sql: should be where.");
            return NULL;
        }
        WhereExprXXCCF = parseExpressionRD(parser);
        if (parser->parserStateType == PARSER_WRONG) {
            return NULL;
        }
    }

    sql_stmt_update *sqlStmtUpdate = (sql_stmt_update *) malloc(sizeof(sql_stmt_update));
    sqlStmtUpdate->tableName = TableNameXXCCF;
    sqlStmtUpdate->fields = FieldsXXCCF;
    sqlStmtUpdate->fieldsExpr = FieldsExprXXCCF;

    TableReference_t *RefXXCCF =   TableReference_make(TableNameXXCCF, NULL);
    SRA_t *TableXXCCF =  SRATable(RefXXCCF);
    sqlStmtUpdate->where = TableXXCCF;
    if (WhereExprXXCCF != NULL) {
        sqlStmtUpdate->where = SRASelect(TableXXCCF, WhereExprXXCCF);
    }
    return sqlStmtUpdate;
};
