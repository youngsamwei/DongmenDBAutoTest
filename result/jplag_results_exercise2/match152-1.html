#include <parser/statement.h>

/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

/*TODO: parse_sql_stmt_update， update语句解析*/

sql_stmt_update *parse_sql_stmt_update(ParserT *parser){

    arraylist *fields=arraylist_create();; //set fields 被更新的字段列表
    arraylist *fieldsExpr=arraylist_create();;  //set fields expression 新值(表达式)列表

    TokenT *token = parseNextToken(parser);

    //判断当前Token是否是“update”和保留字,如果不是，返回Null，如果是吃掉当前Token并获取下一Token
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {
        return NULL;
    }

    //将表名称保存下来，若不是表明，则提示错误信息
    token = parseNextToken(parser);
    char *tableName = NULL;


    if (token->type == TOKEN_WORD) {
        tableName = new_id_name();
        strcpy(tableName, token->text);
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }

    SRA_t *tablesExpr = parseTablesExpr(parser);
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    //判断是否是分号
    token = parseNextToken(parser);
    if (token == NULL || token->type == TOKEN_SEMICOLON) {
        SRA_t *project = SRAProject(tablesExpr, fieldsExpr);
        return NULL;
    }

    //判断当前Token是否是“set”
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "set")) {
        strcpy(parser->parserMessage, "invalid sql: should be set.");
        return NULL;
    }
    token = parseNextToken(parser);
    //将更新字段列表和新值表达式列表提取出来
    if (token->type == TOKEN_WORD) {

        while (token->type == TOKEN_WORD) {
            //获取更新字段
            char *fieldName = new_id_name();
            strcpy(fieldName, token->text);
            arraylist_add(fields, fieldName);
            //吃掉等号
            token = parseEatAndNextToken(parser);
            //获取新值表达式
            token = parseEatAndNextToken(parser);


            Expression *fieldExpr = parseExpressionRD(parser);
            arraylist_add(fieldsExpr,fieldExpr);

            /**
            //如果是字符串
            if (token->type == TOKEN_STRING){

                int len = strlen(token->text) - 1;
                char *v = token->text + 1;
                char *value = (char *) calloc(len, 1);
                strcpy(value, v);
                value[len-1] = '\0';
                arraylist_add(fieldsExpr, value);
                token = parseEatAndNextToken(parser);
            }
            //如果是小数
            else if(token->type ==TOKEN_DECIMAL){
                integer *i = (integer *)calloc(sizeof(integer *), 1);
                i->val = atoi(token->text);
                arraylist_add(fieldsExpr, i);
                token = parseEatAndNextToken(parser);
            }
**/
           /** token = parseEatAndNextToken(parser);
            char *text = new_id_name();
            strcpy(text, token->text);
            printf("%s\n",text);
            if (token->type==TOKEN_COMMA){
                token = parseEatAndNextToken(parser);
            }else{
                break;
            }**/
            /**
            token = parseNextToken(parser);
            char *text = new_id_name();
            strcpy(text, token->text);
            printf("%s\n",text);
             **/
            token = parseNextToken(parser);
            if (token->type==TOKEN_COMMA){
                token = parseEatAndNextToken(parser);
            }else{
                break;
            }
        }
    }

    //获取where表达式
    //token = parseEatAndNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "where")){
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }

    Expression *whereExpr = parseExpressionRD(parser);
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }


    SRA_t *select = SRASelect(tablesExpr, whereExpr);



    sql_stmt_update *sqlStmtUpdate = (sql_stmt_update *)calloc(sizeof(sql_stmt_update),1);

    sqlStmtUpdate->fields=fields;
    sqlStmtUpdate->fieldsExpr=fieldsExpr;
    sqlStmtUpdate->tableName=tableName;
    sqlStmtUpdate->where=select;
    return sqlStmtUpdate;

};
