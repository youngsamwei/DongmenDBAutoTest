#include <parser/statement.h>

/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

/*TODO: parse_sql_stmt_update， update语句解析*/
sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {
    char *tableName = NULL;//列表名
    arraylist *fields = arraylist_create();//字符修改后的值的列表
    arraylist *fieldNames = arraylist_create();//要修改的字段名 fieldsExpr
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }
    TokenT *token = parseNextToken(parser);
    if (token->type == TOKEN_WORD) {
        tableName = new_id_name();//获取列表名
        strcpy(tableName, token->text);
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }

    token = parseEatAndNextToken(parser);
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "set")) {
        strcpy(parser->parserMessage, "invalid sql: should be set.");
        return NULL;
    }
    token = parseNextToken(parser);
    if (token->type == TOKEN_WORD) {
        while (token->type == TOKEN_WORD) {
            char *fieldName = new_id_name();
            strcpy(fieldName, token->text);
            arraylist_add(fields,fieldName);

            token = parseEatAndNextToken(parser);

            if(token->type==TOKEN_EQ){//匹配到等号
                token = parseEatAndNextToken(parser);
            }else{
                return NULL;
            }

            Expression *expr=parseExpressionRD(parser);
            arraylist_add(fieldNames, expr);

            token = parseNextToken(parser);
            if (token!=NULL && token->type==TOKEN_COMMA){//如果后面跟了一个等号，则继续
                token = parseEatAndNextToken(parser);
            }else{
                break;
            }
        }
    } else {
        strcpy(parser->parserMessage, "invalid sql:语法错误.");
        return NULL;
    }

    TableReference_t *ref = TableReference_make(tableName,NULL);
    SRA_t * where = SRATable(ref);
    where -> t = SRA_TABLE;

    sql_stmt_update *sqlStmtUpdate = (sql_stmt_update *)calloc(sizeof(sql_stmt_update),1);
    sqlStmtUpdate->tableName = tableName;
    sqlStmtUpdate->fields = fields;
    sqlStmtUpdate->fieldsExpr = fieldNames;
    sqlStmtUpdate->where=where;
    if(token==NULL)
    {
        return sqlStmtUpdate;
    }
    token = parseNextToken(parser);
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }

    Expression *whereExpr = parseExpressionRD(parser);
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    SRA_t *  table = SRASelect(where,whereExpr);
    table->t = SRA_SELECT;
    sqlStmtUpdate->where=table;
    return sqlStmtUpdate;
};

#include "physicalplan/physicalplan.h"

/*执行 update 语句*/
/*TODO: plan_execute_update， update语句执行*/
int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
    char *tableName=sqlStmtUpdate->tableName;
    arraylist *fields=sqlStmtUpdate->fields; //set fields 被更新的字段列表
    arraylist *fieldsExpr=sqlStmtUpdate->fieldsExpr;  //set fields expression 新值(表达式)列表
    SRA_t *selectStmt=sqlStmtUpdate->where;
    int n=0;
    if (selectStmt != NULL) {
        /*执行select语句，获得物理扫描计划*/
        physical_scan *plan = physical_scan_generate(db, selectStmt, tx);//物理扫描，遍历每一条记录
        arraylist *Name=plan->getFieldsName(plan,tableName);
        plan->beforeFirst(plan);
        while (plan->next(plan)){//若不存在下一条记录，则该循环停止
            n++;
            for (int i = 0; i <fields->size; i++) {
                //对修改的字段进行逐个修改
                char *na=arraylist_get(fields,i);
                variant *v=(variant *)malloc(sizeof(variant)*1);
                memset(v,0, sizeof(v));
                physical_scan_evaluate_expression( arraylist_get(fieldsExpr, i),plan,v);
                if (plan->getField(plan, tableName, na)->type == DATA_TYPE_CHAR )
                {//遍历到的是char类型
                    int MaxSize = plan->getField(plan, tableName, na)->length;
                    int length=strlen(v->strValue);
                    if (MaxSize < length) {
                        v->strValue[MaxSize] = '\0';
                    }
                    plan->setString(plan, tableName, na, v->strValue);
                } else if(plan->getField(plan,tableName,na)->type == DATA_TYPE_INT){
                    int number = v->intValue;
                    plan->setInt(plan, tableName, na, number);
                }
            }
        }
        plan->close(plan);
    }
    return n;
};
