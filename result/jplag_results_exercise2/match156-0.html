char * parseTablesName(ParserT *parser) {
    TokenT *exchange = parseNextToken(parser);
    if (exchange->type == TOKEN_WORD) {
        char *biaoming = new_id_name();
        strcpy(biaoming, exchange->text);
        TokenT *exchange = parseEatAndNextToken(parser);
        return biaoming;
    } else {
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }

}

sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {

    sql_stmt_update *sjkgengxin = (sql_stmt_update *)calloc(sizeof(sql_stmt_update),1);
    TokenT *ToFlag = parseNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }
    ToFlag = parseNextToken(parser);
    char * Tn = parseTablesName(parser);
    sjkgengxin->tableName = Tn;
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    ToFlag = parseNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "set")) {
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }
    ToFlag = parseNextToken(parser);
    arraylist *Fi = arraylist_create();
    arraylist *zhi = arraylist_create();
    while (ToFlag->type == TOKEN_WORD) {
        char *fnme = new_id_name();
        strcpy(fnme, ToFlag->text);
        arraylist_add(Fi, fnme);
        ToFlag = parseEatAndNextToken(parser);
        if(!matchToken(parser, TOKEN_EQ, "=")) {
            strcpy(parser->parserMessage, "invalid sql: incorrect set expression.");
            return NULL;
        }
        Expression *Expr = parseExpressionRD(parser);
        arraylist_add(zhi, Expr);
        ToFlag = parseNextToken(parser);
        if(ToFlag==NULL)
        {
            break;
        }
        if (ToFlag->type == TOKEN_COMMA) {
            parseEatAndNextToken(parser);
            ToFlag = parseNextToken(parser);
        } else {
            break;
        }
    }
    sjkgengxin->fields = Fi;
    sjkgengxin->fieldsExpr = zhi;
    ToFlag = parseNextToken(parser);
    if (ToFlag == NULL || ToFlag->type == TOKEN_SEMICOLON) {
        TableReference_t *ref =   TableReference_make(Tn, NULL);
        SRA_t *biao =  SRATable(ref);
        sjkgengxin->where = biao;
        return sjkgengxin;
    }
    ToFlag = parseNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }
    ToFlag = parseNextToken(parser);
    Expression *wExpr = parseExpressionRD(parser);
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    ToFlag = parseNextToken(parser);
    if (ToFlag == NULL || ToFlag->type == TOKEN_SEMICOLON) {
        TableReference_t *ref =   TableReference_make(Tn, NULL);
        SRA_t *biao =  SRATable(ref);
        SRA_t *slt = SRASelect(biao, wExpr);
        sjkgengxin->where = slt;
        return sjkgengxin;
    }
};

int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate, transaction *tx){
    int sum=0;
    physical_scan *todo = physical_scan_generate(db, sqlStmtUpdate->where, tx);
    todo->beforeFirst(todo);
    char * task = sqlStmtUpdate->tableName;
    arraylist * FFFFFFFa = sqlStmtUpdate->fields;
    while (todo->next(todo)) {
        for(int cnt = 0; cnt < FFFFFFFa->size; cnt++) {
            char * Fi = arraylist_get(FFFFFFFa, cnt);
            variant * flag = malloc(sizeof(variant));
            enum data_type field_type = todo->getField(todo,task,Fi)->type;

            physical_scan_evaluate_expression(arraylist_get(sqlStmtUpdate->fieldsExpr, cnt), todo, flag);
            if(flag->type!=field_type)
            {
                fprintf(stdout,"error");
                return DONGMENDB_EINVALIDSQL;
            }
            if(flag->type == DATA_TYPE_INT) {
                todo->setInt(todo, task, Fi, flag->intValue);
            }
            else if(flag->type==DATA_TYPE_CHAR) {
                todo->setString(todo, task, Fi, flag->strValue);
            }
        }
        sum++;
    }
    todo->close(todo);
    return sum;
    // return DONGMENDB_OK;

};
