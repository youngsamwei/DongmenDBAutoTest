char * parseTablesName(ParserT *parser) {
    TokenT *token = parseNextToken(parser);
    if (token->type == TOKEN_WORD) {
        char *tableName = new_id_name();
        strcpy(tableName, token->text);
        TokenT *token = parseEatAndNextToken(parser);
        return tableName;
    } else {
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }

}

sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {

    sql_stmt_update *sqlStmtUpdate = (sql_stmt_update *)calloc(sizeof(sql_stmt_update),1);
    TokenT *token = parseNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }
    token = parseNextToken(parser);
    char * tableName = parseTablesName(parser);
    sqlStmtUpdate->tableName = tableName;
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    token = parseNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "set")) {
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }
    token = parseNextToken(parser);
    arraylist *fields = arraylist_create();
    arraylist *values = arraylist_create();
    while (token->type == TOKEN_WORD) {
        char *fieldName = new_id_name();
        strcpy(fieldName, token->text);
        arraylist_add(fields, fieldName);
        token = parseEatAndNextToken(parser);
        if(!matchToken(parser, TOKEN_EQ, "=")) {
            strcpy(parser->parserMessage, "invalid sql: incorrect set expression.");
            return NULL;
        }
        Expression *Expr = parseExpressionRD(parser);
        arraylist_add(values, Expr);
        token = parseNextToken(parser);
        if(token==NULL)
        {
            break;
        }
        if (token->type == TOKEN_COMMA) {
            parseEatAndNextToken(parser);
            token = parseNextToken(parser);
        } else {
            break;
        }
    }
    sqlStmtUpdate->fields = fields;
    sqlStmtUpdate->fieldsExpr = values;
    token = parseNextToken(parser);

    if (token == NULL || token->type == TOKEN_SEMICOLON) {
        TableReference_t *ref =   TableReference_make(tableName, NULL);
        SRA_t *table =  SRATable(ref);
//        SRA_t *project = SRASelect(table, NULL);
        sqlStmtUpdate->where = table;
        return sqlStmtUpdate;
    }
    token = parseNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "where")){
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }
    token = parseNextToken(parser);
    Expression *whereExpr = parseExpressionRD(parser);
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    token = parseNextToken(parser);
    if (token == NULL || token->type == TOKEN_SEMICOLON) {
        TableReference_t *ref =   TableReference_make(tableName, NULL);
        SRA_t *table =  SRATable(ref);
        SRA_t *select = SRASelect(table, whereExpr);
        sqlStmtUpdate->where = select;
        return sqlStmtUpdate;
    }
};

int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
    int count=0;
    physical_scan *plan = physical_scan_generate(db, sqlStmtUpdate->where, tx);
    plan->beforeFirst(plan);
    char * tableName = sqlStmtUpdate->tableName;
    arraylist * fields = sqlStmtUpdate->fields;
    while (plan->next(plan)){
        for(int i = 0;i < fields->size; i++){
            char * fieldName = arraylist_get(fields, i);
            variant * var = malloc(sizeof(variant));
            enum data_type field_type = plan->getField(plan,tableName,fieldName)->type;

            physical_scan_evaluate_expression(arraylist_get(sqlStmtUpdate->fieldsExpr, i), plan, var);
            if(var->type!=field_type)
            {
                fprintf(stdout,"error");
                return DONGMENDB_EINVALIDSQL;
            }
            if(var->type == DATA_TYPE_INT){
                plan->setInt(plan, tableName, fieldName, var->intValue);
            }
            else if(var->type==DATA_TYPE_CHAR){
                plan->setString(plan, tableName, fieldName, var->strValue);
            }
        }
        count++;
    }
    plan->close(plan);
    return  count;
   // return DONGMENDB_OK;

};
