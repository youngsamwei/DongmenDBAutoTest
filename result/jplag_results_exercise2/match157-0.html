#include <parser/statement.h>

/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */
/*TODO: parse_sql_stmt_update， update语句解析*/
sql_stmt_update *parse_sql_stmt_update(ParserT *pa) {

    /*char *tableName = NULL;
    arraylist *fields = arraylist_create();
    arraylist *fieldsExpr = arraylist_create();*/

    sql_stmt_update *s = (sql_stmt_update *) malloc(sizeof(sql_stmt_update));
    if (!matchToken(pa, TOKEN_RESERVED_WORD, "update")) {
        strcpy(pa->parserMessage, "no update.");
        return NULL;
    }
    TokenT *to = parseNextToken(pa);
    to = parseNextToken(pa);
    if (to->type == TOKEN_WORD)
        strcpy(s->tableName, to->text);
    else {
        strcpy(pa->parserMessage, "no table name.");
        return NULL;
    }

    to = parseEatAndNextToken(pa);
    if (!matchToken(pa, TOKEN_RESERVED_WORD, "set")) {
        strcpy(pa->parserMessage, "no set.");
        return NULL;
    }

    to = parseNextToken(pa);

    while (to != NULL ) {
        if(to->type == TOKEN_WORD || to->type == TOKEN_COMMA) {
            if (to->type == TOKEN_COMMA) {
                /*若是逗号，则需要跳过逗号*/
                to = parseEatAndNextToken(pa);
            }
            /*获取字段名*/
            if (to->type == TOKEN_WORD) {
                char *fn = new_id_name();
                strcpy(fn, to->text);
                arraylist_add(s->fields, fn);

            } else {
                strcpy(pa->parserMessage, "no project name.");
                return NULL;
            }

            to = parseEatAndNextToken(pa);
            if (!matchToken(pa, TOKEN_EQ, "=")) {
                strcpy(pa->parserMessage, "no =.");
                return NULL;
            }

            Expression *expr = parseExpressionRD(pa);
            arraylist_add(s->fieldsExpr, expr);

            to = parseNextToken(pa);
            if (to == NULL)
                break;
        }
    }

    Expression *wh = NULL;
    if (to != NULL) {
        if (!matchToken(pa, TOKEN_RESERVED_WORD, "where")) {
            strcpy(pa->parserMessage, "no where.");
            return NULL;
        }
        wh = parseExpressionRD(pa);
        if (pa->parserStateType == PARSER_WRONG)
            return NULL;
    }



    TableReference_t *ref = TableReference_make(s->tableName, NULL);
    SRA_t *table = SRATable(ref);
    s->where = table;
    if (wh != NULL)
        s->where = SRASelect(table, wh);

    return s;
}

#include "physicalplan/physicalplan.h"

/*执行 update 语句*/
/*TODO: plan_execute_update， update语句执行*/
int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx) {
    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_update 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成update操作
     * */
    char *tn = sqlStmtUpdate->tableName;
    arraylist *f = sqlStmtUpdate->fields; //set fields 被更新的字段列表
    arraylist *fe = sqlStmtUpdate->fieldsExpr;  //set fields expression 新值(表达式)列表
    SRA_t *w = sqlStmtUpdate->where;
    physical_scan *p = physical_scan_generate(db, w, tx);
    p->beforeFirst(p);
    variant *nowv = (variant *) malloc(sizeof(variant));
    while ((p->next(p))!=NULL) {

        for (size_t i = 0; i < f->size; i++) {
            char *n = arraylist_get(f, i);
            Expression *ex = arraylist_get(fe, i);
            physical_scan_evaluate_expression(ex, p, nowv);
            if (nowv->type = DATA_TYPE_CHAR) {
                p->setString(p, tn, n, nowv->strValue);
            } else if (nowv->type = DATA_TYPE_INT) {
                p->setInt(p, tn, n, nowv->intValue);
            } else {
                return DONGMENDB_EINVALIDSQL;
            }
        }
    }
    return DONGMENDB_OK;
};
