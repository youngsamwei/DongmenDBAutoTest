int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx) {
    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_update 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成update操作
     * */
    char *kn = sqlStmtUpdate->tableName;
    arraylist *f = sqlStmtUpdate->fields; //set fields 被更新的字段列表
    arraylist *fe = sqlStmtUpdate->fieldsExpr;  //set fields expression 新值(表达式)列表
    SRA_t *w = sqlStmtUpdate->where;
    physical_scan *l = physical_scan_generate(db, w, tx);
    l->beforeFirst(l);
    variant *nowv = (variant *) malloc(sizeof(variant));
    while ((l->next(l))!=NULL) {

        for (size_t i = 0; i < f->size; i++) {
            char *m = arraylist_get(f, i);
            Expression *e = arraylist_get(fe, i);
            physical_scan_evaluate_expression(e, l, nowv);
            if (nowv->type = DATA_TYPE_CHAR) {
                l->setString(l, kn, m, nowv->strValue);
            } else if (nowv->type = DATA_TYPE_INT) {
                l->setInt(l, kn, m, nowv->intValue);
            } else {
                return DONGMENDB_EINVALIDSQL;
            }
        }
    }
    return DONGMENDB_OK;
};

#include <parser/statement.h>

/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */
/*TODO: parse_sql_stmt_update， update语句解析*/

sql_stmt_update *parse_sql_stmt_update(ParserT *po) {

    /*char *tableName = NULL;
    arraylist *fields = arraylist_create();
    arraylist *fieldsExpr = arraylist_create();*/

    sql_stmt_update *s = (sql_stmt_update *) malloc(sizeof(sql_stmt_update));
    if (!matchToken(po, TOKEN_RESERVED_WORD, "update")) {
        strcpy(po->parserMessage, "no update.");
        return NULL;
    }
    TokenT *t = parseNextToken(po);
    t = parseNextToken(po);
    if (t->type == TOKEN_WORD)
        strcpy(s->tableName, t->text);
    else {
        strcpy(po->parserMessage, "no table name.");
        return NULL;
    }

    t = parseEatAndNextToken(po);
    if (!matchToken(po, TOKEN_RESERVED_WORD, "set")) {
        strcpy(po->parserMessage, "no set.");
        return NULL;
    }

    t = parseNextToken(po);

    while (t != NULL ) {
        if(t->type == TOKEN_WORD || t->type == TOKEN_COMMA) {
            if (t->type == TOKEN_COMMA) {
                /*若是逗号，则需要跳过逗号*/
                t = parseEatAndNextToken(pa);
            }
            /*获取字段名*/
            if (t->type == TOKEN_WORD) {
                char *f = new_id_name();
                strcpy(f, t->text);
                arraylist_add(s->fields, f);

            } else {
                strcpy(po->parserMessage, "no project name.");
                return NULL;
            }

            t = parseEatAndNextToken(po);
            if (!matchToken(po, TOKEN_EQ, "=")) {
                strcpy(po->parserMessage, "no =.");
                return NULL;
            }

            Expression *expr = parseExpressionRD(po);
            arraylist_add(s->fieldsExpr, expr);

            t = parseNextToken(po);
            if (t == NULL)
                break;
        }
    }

    Expression *w = NULL;
    if (t != NULL) {
        if (!matchToken(po, TOKEN_RESERVED_WORD, "where")) {
            strcpy(po->parserMessage, "no where.");
            return NULL;
        }
        wh = parseExpressionRD(po);
        if (po->parserStateType == PARSER_WRONG)
            return NULL;
    }



    TableReference_t *ref = TableReference_make(s->tableName, NULL);
    SRA_t *table = SRATable(ref);
    s->where = table;
    if (w != NULL)
        s->where = SRASelect(table, w);

    return s;
}
