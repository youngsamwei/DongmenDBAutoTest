 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *UPDATE Person SET Address = 'Zhongshan 23', City = 'Nanjing'
 *WHERE LastName = 'Wilson'
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

sql_stmt_update *parse_sql_stmt_update(ParserT *parser){
    // fprintf(stderr, "TODO: update is not implemented yet. in parse_sql_stmt_update \n");
    char *tbN = NULL;//char *tableName = NULL;
    arraylist *fls = arraylist_create();//arraylist *fields = arraylist_create();
    arraylist *flsE = arraylist_create();

    // * 首先匹配 update, 使用matchToken(parser,TOKEN_RESERVED_WORD,"update")匹配
    TokenT *tk_key = parseNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "update")){
        strcpy(parser->parserMessage, "error1.");
        return NULL;
    }

    // * 获取表名 使用 if 判断是否是 TOKEN_WORD 类型，如果是，获得表名；给字符串指针开空间的时候可以使用new_id_name()函数
    tk_key = parseNextToken(parser);
    if(tk_key->type == TOKEN_WORD){
        tbN = new_id_name();
        strcpy(tbN,tk_key->text);
        //token = parseEatToken(parser);
    }else {
        strcpy(parser->parserMessage,"error2.");
        return  NULL;
    }

    // * 匹配 set 使用 matchToken(parser,TOKEN_RESERVED_WORD,"set")
    tk_key = parseEatAndNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "set")){
        strcpy(parser->parserMessage, "error3.");
        return NULL;
    }

    // * 使用循环获得字段名（TOKEN_WORD 类型），放入一个 fields(arraylist 类型)中
    // * 循环获得值（或者表达式）parseExpressionRD(parse)函数,fieldsExpr(arraylist 类型)中
    // update student
    //set sage=19,semail="1291332"
    //where sname='张三'
    tk_key = parseNextToken(parser);
    if(tk_key->type == TOKEN_WORD) {

        while(tk_key->type == TOKEN_WORD ) {
            //列名
            arraylist_add(fls,tk_key->text);
            //等号
            tk_key = parseEatAndNextToken(parser);
            if (tk_key->type == TOKEN_EQ) {
                tk_key = parseEatAndNextToken(parser);
            } else {
                strcpy(parser->parserMessage, "error4.");
                return NULL;
            }
            //值
//            if (token->type == TOKEN_STRING || token->type == TOKEN_DECIMAL) {
//
//                if (token->type == TOKEN_STRING) {
//                    /*去掉引号, 已经在tokenizer.c中完成*/
//                    int len = strlen(token->text) - 1;
//                    char *v = token->text + 1;
//                    char *value = (char *) calloc(len, 1);
//                    strcpy(value, v);
//                    value[len - 1] = '\0';
//
//                    arraylist_add(fieldsExpr, token->text);
//                } else {
//                        integer *i = (integer *) calloc(sizeof(integer *), 1);
//                        i->val = atoi(token->text);
//                        arraylist_add(fieldsExpr, i);
//                }
//            } else {
//                strcpy(parser->parserMessage, "语法错误.");
//                return NULL;
//            }
            Expression *expression = parseExpressionRD(parser);
            arraylist_add(flsE,expression);

            //逗号
            tk_key = parseNextToken(parser);
            if(tk_key == NULL){
                break;
            }
            if(tk_key->type == TOKEN_COMMA){
                tk_key = parseEatAndNextToken(parser);
            }else {
                break;
            }

        }

    }else  if(tk_key->type != TOKEN_WORD)  {
        strcpy(parser->parserMessage, "error5.");
        return NULL;
    }

    // * 匹配 where，matchToken(parser,TOKEN_RESERVED_WORD,"where")，注意不是所有update语句都伴随一个where表达式
    // * 使用parseExpressionRD(parse)获取值（或者表达式），并包裹在SRA_t中（使用SRA_SELECT构造），作为sql_stmt_update->where的值。
    // *（语句若不伴随where表达式则需要以一个SRA_TABLE代替，具体实现可参考 exp_01_stmt_parser/exp_01_03_select.c）
    //where
    tk_key = parseNextToken(parser);
    /*if(!matchToken(parser,TOKEN_RESERVED_WORD,"where")){
        strcpy(parser->parserMessage,"error6");
        return  NULL;
    }*/

    //表名为空
    TableReference_t *ref1_sh = TableReference_make(tbN,NULL);
    SRA_t *where_sh = SRATable(ref1_sh);
    if(matchToken(parser,TOKEN_RESERVED_WORD,"where")) {

        //where条件
        tk_key = parseNextToken(parser);
        Expression *whereExpr_sh = parseExpressionRD(parser);
        where_sh = SRASelect(where_sh, whereExpr_sh);
        if (parser->parserStateType == PARSER_WRONG) {
            return NULL;
        }

    }
    // * 创建sql_stmt_update指针，开空间，对各字段进行赋值，返回sql_stmt_update结构体
    sql_stmt_update *SqlStmtUpdate_sh = (sql_stmt_update *)calloc(sizeof(sql_stmt_update),1);
    SqlStmtUpdate_sh->tableName = tbN;
    SqlStmtUpdate_sh->fields = fls;
    SqlStmtUpdate_sh->fieldsExpr = flsE;
    SRA_print(where_sh);
    SqlStmtUpdate_sh->where = where_sh;

    return  SqlStmtUpdate_sh;

};

int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_update 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成update操作
     * */
    //fprintf(stderr, "TODO: update is not implemented yet. in plan_execute_update \n");

    //记录被更新的条数
    size_t updated_sum = 0;

//产生物理计划
    physical_scan *ph_sc = physical_scan_generate(db,sqlStmtUpdate->where,tx);
//遍历计划（循环）
    //置于最前
    ph_sc->beforeFirst(ph_sc);
    //循环扫描，查找有没有下一个
    while(ph_sc->next(ph_sc)){
        //针对每个字段进行处理，进行循环，字段判断条件
        for(size_t i = 0 ; i < sqlStmtUpdate->fields->size ; i++){
            //获取第i个字段的名字
            char *curFieldName_sh = arraylist_get(sqlStmtUpdate->fields , i);
            //创建可变结构体,存储set等号后面的表达式值
            variant *var_sh = (variant*)calloc(sizeof(variant),1);
            //获取字段的数据类型
            enum data_type fields_type_sh = ph_sc->getField(ph_sc , sqlStmtUpdate->tableName , curFieldName_sh)->type;
            //获取表达式值放到可变结构体，并计算赋值表达式
            physical_scan_evaluate_expression(arraylist_get(sqlStmtUpdate->fieldsExpr , i) , ph_sc , var_sh);
            //如果与当前数据类型不匹配
            if(var_sh->type != fields_type_sh){
                fprintf(stdout , "error");
                return DONGMENDB_EINVALIDSQL;
            }
            //如果与当前数据类型匹配
            if(var_sh->type == DATA_TYPE_INT){
                //为INT类型，则使用scan设置int值
                ph_sc->setInt(ph_sc , sqlStmtUpdate->tableName , curFieldName_sh , var_sh->intValue);
            }else if(var_sh->type == DATA_TYPE_CHAR){
                //为CHAR类型，则使用scan设置char值
                ph_sc->setString(ph_sc , sqlStmtUpdate->tableName , curFieldName_sh , var_sh->strValue);
            }
        }
        //条数增加
        updated_sum++;
    }
//关闭scan
    ph_sc->close(ph_sc);
    return updated_sum;
};
