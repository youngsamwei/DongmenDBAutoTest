#include <parser/statement.h>
#include <parser/expression.h>

/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

/*TODO: parse_sql_stmt_update， update语句解析*/

sql_stmt_update *parse_sql_stmt_update(ParserT *parser){
    //fprintf(stderr, "TODO: update is not implemented yet. in parse_sql_stmt_update \n");
    arraylist *fieldNa=arraylist_create();
    arraylist *fieldex=arraylist_create();
    sql_stmt_update *update=malloc(sizeof(sql_stmt_update));


    TokenT *token=parseNextToken(parser);

    //update
    if(!matchToken(parser,TOKEN_RESERVED_WORD,"update")){
        return NULL;
    }
    token=parseNextToken(parser);


    //tablename
    if(token->type==TOKEN_WORD) {
        update->tableName = new_id_name();
        strcpy(update->tableName, token->text);
        token=parseEatAndNextToken(parser);
    }else{
        strcpy(parser->parserMessage,"invalid sql:missing table name.");
        return NULL;
    }

    //别名为空
    TableReference_t *r1 =   TableReference_make(update->tableName, NULL);
    SRA_t *where =  SRATable(r1);


    //set
    if(!matchToken(parser,TOKEN_RESERVED_WORD,"set"))
    {
        strcpy(parser->parserMessage,"invalid sql: missing set.");
        return  NULL;
    }

    //解析字段,直到遇到where保留字
    token=parseNextToken(parser);
    while(token->type ==TOKEN_WORD) {

        //fieldname
        arraylist_add(fieldNa,token->text);
        token=parseEatAndNextToken(parser);


        if(!matchToken(parser,TOKEN_EQ,"="))
        {
            strcpy(parser->parserMessage,"invalid sql:missing =!");
            return NULL;
        }

        Expression *ex=parseExpressionRD(parser);
        arraylist_add(fieldex,ex);
        token=parseNextToken(parser);

        if(token!=NULL)
        {
            if(matchToken(parser,TOKEN_COMMA,","))
            {
                token=parseNextToken(parser);
                continue;
            }
        }
        else
        {
            break;//到达语句末尾
        }

        token=parseNextToken(parser);

    }


    Expression *whereEx;
    if(token!=NULL&&matchToken(parser,TOKEN_RESERVED_WORD,"where")){
        //strcpy(parser->parserMessage,"invalid sql:missing where.");
        //return NULL;
        token = parseNextToken(parser);
        whereEx = parseExpressionRD(parser); //where 后面的条件表达式
        where=SRASelect(where,whereEx);
        if (parser->parserStateType == PARSER_WRONG) {
            return NULL;
        }
    }




    update->fields=fieldNa;
    update->fieldsExpr=fieldex;
    SRA_print(where);
    update->where=where;
    return update;

};

int plan_execute_update(dongmendb *db, sql_stmt_update *sqlUpdate , transaction *t) {


    //fprintf(stderr, "TODO: update is not implemented yet. in plan_execute_update \n");
    physical_scan *plan1 = physical_scan_generate(db, sqlUpdate->where, t);
    int update_lines=0;

    plan1->beforeFirst(plan1);

    while(plan1->next(plan1))
    {

        for(size_t i=0;i<sqlUpdate->fields->size;i++){
            char *fieldna=arraylist_get(sqlUpdate->fields,i);
            variant *var=malloc(sizeof(variant));
            physical_scan_evaluate_expression(arraylist_get(sqlUpdate->fieldsExpr,i),plan1,var);
            if(var->type==DATA_TYPE_INT){
                plan1->setInt(plan1,sqlUpdate->tableName,fieldna,var->intValue);
            }
            else if(var->type==DATA_TYPE_CHAR){
                plan1->setString(plan1,sqlUpdate->tableName,fieldna,var->strValue);
            }
        }
        update_lines++;
    }
    plan1->close(plan1);




    return update_lines;
};
