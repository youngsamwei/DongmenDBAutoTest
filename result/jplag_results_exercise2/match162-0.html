/*执行 update 语句的物理计划，返回修改的记录条数
 * 返回大于等于0的值，表示修改的记录条数；
 * 返回小于0的值，表示修改过程中出现错误。
 * */
/*TODO: plan_execute_update， update语句执行*/
int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_update 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成update操作
     * */
   // fprintf(stderr, "TODO: update is not implemented yet. in plan_execute_update \n");
    arraylist *fields = sqlStmtUpdate->fields;
    arraylist *fieldsExpr = sqlStmtUpdate->fieldsExpr;
    char *tablename = sqlStmtUpdate->tableName;
    SRA_t *where = sqlStmtUpdate->where;
    int num=0;
//构造物理计划
    physical_scan *plan = physical_scan_generate(db, where, tx);
    plan->beforeFirst(plan);
    while(plan->next(plan)){
        for (int i = 0; i < fields->size; i++) {
            variant *vars = (variant *) calloc(sizeof(variant), 1);
            physical_scan_evaluate_expression(arraylist_get(fieldsExpr, i), plan, vars);
            char *fieldName = arraylist_get(fields, i);
            int type = vars->type;
            if (type == DATA_TYPE_INT) {
                plan->setInt(plan, tablename, fieldName, vars->intValue);
            } else if (type == DATA_TYPE_CHAR) {
                void_ptr *ptr = (void_ptr *) calloc(sizeof(void_ptr *), 1);
                hashmap_get(plan->physicalScanSelect->scan->physicalScanTable->tableInfo->fields, fieldName, ptr);
                field_info *fieldInfo = *ptr;
                char *val = vars->strValue;
                if (fieldInfo->length < strlen(val)) {
                    val[fieldInfo->length] = '\0';
                }
                plan->setString(plan, tablename, fieldName, val);
            } else {
                return DONGMENDB_EINVALIDSQL;
            }
        }printf("\n");
        num++;
    }
    plan->close(plan);
    return num;
};

#include <parser/statement.h>
#include <parser/parser.h>
#include <utils/utils.h>
/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

/*TODO: parse_sql_stmt_update， update语句解析*/
sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {
//    fprintf(stderr, "TODO: update is not implemented yet. in parse_sql_stmt_update \n");
    char *tableName = NULL;
    sql_stmt_update *xj = (sql_stmt_update *) calloc(sizeof(sql_stmt_update), 1);
    arraylist *fields = arraylist_create();
    arraylist *fieldsExpr = arraylist_create();

    if (!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {
        return NULL;
    }//判断第一个词是不是update
    TokenT *token = parseNextToken(parser);
    SRA_t *tablesExpr = parseTablesExpr(parser);
    if (token->type == TOKEN_WORD) {
        tableName = new_id_name();
        strcpy(tableName, token->text);
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }//判断是不是表名，不是的话就保留错误信息
    token = parseNextToken(parser);
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "set")) {
        return NULL;
    }//判断是不是set
    //匹配set后面的字段
    token = parseNextToken(parser);
    while (true) {
        token = parseNextToken(parser);

        if (token->type == TOKEN_WORD) {
            char *fieldName = new_id_name();
            strcpy(fieldName, token->text);
            arraylist_add(fields, fieldName);
            parser->currToken = TKGetNextToken(parser->tokenizer);
            token = parseNextToken(parser);
        } else {
            strcpy(parser->parserMessage, "defete");
        }
        if (token->type == TOKEN_EQ) {
            parser->currToken = TKGetNextToken(parser->tokenizer);
            token = parseNextToken(parser);
        } else {
            return NULL;
        }


        Expression *expr1 = parseExpressionRD(parser);
        arraylist_add(fieldsExpr, expr1);
        token = parseNextToken(parser);
        if (parser->currToken == NULL) {
            parser->currToken = token;//为了防止空指针 没有where currtoken会不受影响的指针下移
            xj->fields = fields;
            xj->fieldsExpr = fieldsExpr;
            xj->tableName = tableName;
            TableReference_t *re = TableReference_make(tableName, tableName);
            SRA_t *whereone = SRATable(re);
            xj->where = whereone;
            return xj;
        }

        if (token->type == TOKEN_COMMA) {
            parser->currToken = TKGetNextToken(parser->tokenizer);
            token = parseNextToken(parser);
            continue;
        } else {
            if (matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
                break;
            }
        }
    }
    token = parseNextToken(parser);
    Expression *whereExpr = parseExpressionRD(parser);
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    SRA_t *where = SRASelect(tablesExpr, whereExpr);
    xj->fields = fields;
    xj->fieldsExpr = fieldsExpr;
    xj->tableName = tableName;
    xj->where = where;
    return xj;
};
