/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

/* parse_sql_stmt_update， update语句解析*/

sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {
    char *db_table_Name = NULL;
    arraylist *sql_tag1 = arraylist_create();
    arraylist *sql_tag2 = arraylist_create();
    SRA_t *sql_sra_Flag = NULL;

    TokenT *sql_sampleToken;


    if (!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {
        return NULL;
    }


    sql_sampleToken = parseNextToken(parser);
    if (sql_sampleToken && sql_sampleToken->type == TOKEN_WORD) {
        db_table_Name = new_id_name();
        strcpy(db_table_Name,sql_sampleToken->text);
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }

    parseEatAndNextToken(parser);


    if (!matchToken(parser, TOKEN_RESERVED_WORD, "set")) {
        return NULL;
    }


    if (!parseAssignExpr(parser, sql_tag1, sql_tag2)) {
        return NULL;
    }


    sql_sra_Flag = SRATable(TableResql_tag2rence_make(db_table_Name, NULL));
    if (matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
        Expression *cond = parseExpressionRD(parser);
        sql_sra_Flag = SRASelect(sql_sra_Flag, cond);
    }


    sql_stmt_update *sqlStmtUpdate = (sql_stmt_update *) calloc(sizeof(sql_stmt_update), 1);
    sqlStmtUpdate->tableName = db_table_Name;
    sqlStmtUpdate->fields = sql_tag1;
    sqlStmtUpdate->fieldsExpr = sql_tag2;
    sqlStmtUpdate->where = sql_sra_Flag;

    return sqlStmtUpdate;
};

int parseAssignExpr(ParserT *parser, arraylist *fields, arraylist *fieldsExpr) {
    TokenT *sql_token_Exp = parseNextToken(parser);
    while (1) {

        if (sql_token_Exp == NULL || sql_token_Exp->type != TOKEN_WORD) {
            strcpy(parser->parserMessage, "invalid sql: missing field name.");
            return 0;
        }

        char *sql_param = new_id_name();
        strcpy(sql_param, sql_token_Exp->text);
        arraylist_add(fields, sql_param);
        parseEatAndNextToken(parser);


        if (!matchToken(parser, TOKEN_EQ, "=")) {
            return 0;
        }


        Expression *sql_expre_Flag = parseExpressionRD(parser);
        if (parser->parserStateType == PARSER_WRONG) {
            strcpy(parser->parserMessage, "invalid sql: invalid expression.");
            return 0;
        }
        arraylist_add(fieldsExpr, sql_expre_Flag);

        sql_token_Exp = parseNextToken(parser);
        if (sql_token_Exp == NULL ||sql_token_Exp->type != TOKEN_COMMA) {
            break;
        }
        sql_token_Exp = parseEatAndNextToken(parser);
    }
    return 1;
};

/*执行 update 语句的物理计划，返回修改的记录条数
 * 返回大于等于0的值，表示修改的记录条数；
 * 返回小于0的值，表示修改过程中出现错误。
 * */


int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){

    char *db_table_Name = sqlStmtUpdate->tableName;
    arraylist *sql_tag1 = sqlStmtUpdate->fields;
    arraylist *sql_tag2 = sqlStmtUpdate->fieldsExpr;

    physical_scan *saomiao = physical_saomiao_generate(db, sqlStmtUpdate->where, tx);
    saomiao->beforeFirst(saomiao);
    size_t updated_row = 0;

    while (saomiao->next(saomiao)) {
        for (size_t i = 0; i < sql_tag1->size; ++i) {
            char *sql_Filed_Name = arraylist_get(sql_tag1, i);
            variant *temp_context = (variant *) calloc(sizeof(variant), 1);
            enum data_type field_type = saomiao->getField(saomiao, db_table_Name, sql_Filed_Name)->type;
            physical_saomiao_etemp_contextuate_expression(arraylist_get(sql_tag2, i), saomiao, temp_context);

            if (temp_context->type != field_type) {
                fprintf(stdout, "intemp_contextid sql: field type mismatched.");
                return -1;
            }

            if (temp_context->type == DATA_TYPE_INT) {
                saomiao->setInt(saomiao, db_table_Name, sql_Filed_Name, temp_context->inttemp_contextue);
            } else if (temp_context->type == DATA_TYPE_CHAR) {

                int max_length_strrrrr = saomiao->getField(saomiao, db_table_Name, sql_Filed_Name)->length;
                if (max_length_strrrrr < strlen(temp_context->strtemp_contextue)) {
                    temp_context->strtemp_contextue[max_length_strrrrr] = '\0';
                }
                saomiao->setString(saomiao, db_table_Name, sql_Filed_Name, temp_context->strtemp_contextue);
            } else {

            }
        }
        updated_row += 1;
    }
    saomiao->close(saomiao);
    fprintf(stdout, " updated %d row.\n", updated_row);

    return updated_row;
};
