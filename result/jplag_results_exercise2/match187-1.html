/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

/*TODO: parse_sql_

sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {

    char *name;
    arraylist *fie=arraylist_create();
    arraylist *Expr=arraylist_create();
    SRA_t *wh=NULL;

    TokenT *tok = parseNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "update")){
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;  //pipei update
    }
    //表名
    tok=parseNextToken(parser);
    if(tok->type==TOKEN_WORD)
    {
        name=new_id_name();
        strcpy(name,tok->text);
    }
    tok = parseEatAndNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "set")){
        strcpy(parser->parserMessage, "语法错误.");
        return NULL; //pipei set
    }
    while(1){
        tok=parseNextToken(parser);
        if(tok->type==TOKEN_WORD){
            char* fn=new_id_name();
            strcpy(fn,tok->text);
            arraylist_add(fie,fn);
        }
        tok=parseEatAndNextToken(parser);
        if(!matchToken(parser,TOKEN_EQ,"="))
        {
            return NULL;
        }
        Expression* asd=parseExpressionRD(parser);
        arraylist_add(Expr,asd);
        tok=parseNextToken(parser);

        if(tok==NULL||tok->type==TOKEN_RESERVED_WORD)//gengxin
            break;

        tok=parseEatAndNextToken(parser);//chidiao
    }
    Expression* wc=NULL;

    if(tok!=NULL&&matchToken(parser,TOKEN_RESERVED_WORD,"where")){
        wc=parseExpressionRD(parser);
    }
    TableReference_t *rt=TableReference_make(name,NULL);
    wh=SRATable(rt);  //pipei where

    if(wc!=NULL){
        wh=SRASelect(wh,wc);
    }
    sql_stmt_update* su=(sql_stmt_update*)malloc(sizeof(sql_stmt_update));

    su->tableName=name;
    su->where=wh;
    su->fields=fie;
    su->fieldsExpr=Expr;
    return su;
}

int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_update 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成update操作
     * */

    //fprintf(stderr, "TODO: update is not implemented yet. in plan_execute_update \n");
    //创建一个物理扫描计划
    physical_scan* haha=physical_scan_generate(db,sqlStmtUpdate->where,tx);
    //初始化扫描计划
    haha->beforeFirst(haha);

    //遍历所有条目

    int zsh=0;
    while(haha->next(haha)){
        zsh++;
        variant* hehe=malloc(sizeof(variant));
        for(size_t sb=0;sb<sqlStmtUpdate->fieldsExpr->size;sb++){
            //计算表达式的值，根据值类型，分别调用setint setstring
            physical_scan_evaluate_expression(arraylist_get(sqlStmtUpdate->fieldsExpr,sb),haha,hehe);
            if(hehe->type==DATA_TYPE_INT){
                haha->setInt(haha,sqlStmtUpdate->tableName,arraylist_get(sqlStmtUpdate->fields,sb),hehe->intValue);
            }
            else if(hehe->type==DATA_TYPE_CHAR){
                haha->setString(haha,sqlStmtUpdate->tableName,arraylist_get(sqlStmtUpdate->fields,sb),hehe->strValue);
            }

        }

    }
    //关闭扫描计划
    haha->close(haha);
    return zsh;
   // return DONGMENDB_OK;

};
