/*执行 update 语句*/
/*TODO: plan_execute_update， update语句执行*/
int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_update 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成update操作
     * */
    int flag=0;
    physical_scan *scan = physical_scan_generate(db, sqlStmtUpdate->where, tx);//创建全表扫描计划
    scan->beforeFirst(scan);//初始化
    while (scan->next(scan)) {//当还有扫描计划时
        for (size_t i = 0; i < sqlStmtUpdate->fields->size; ++i) {  //arraylist *fields; //set fields 被更新的字段列表
            char *FieldName = arraylist_get(sqlStmtUpdate->fields, i);//获取要更新的列名
            Expression *currentFieldsExprName=arraylist_get(sqlStmtUpdate->fieldsExpr,i);//获取需要更新的值的表达式
            variant *tem_val = (variant *) calloc(sizeof(variant), 1);//保存需要更新的字段类型
            enum data_type field_type = scan->getField(scan, sqlStmtUpdate->tableName, FieldName)->type; //获取列名的的类型
            physical_scan_evaluate_expression(currentFieldsExprName, scan, tem_val);//计算后序表达式的值
            if (tem_val->type == DATA_TYPE_INT) {
                scan->setInt(scan, sqlStmtUpdate->tableName, FieldName, tem_val->intValue);
            } else if (tem_val->type == DATA_TYPE_CHAR) {
                scan->setString(scan, sqlStmtUpdate->tableName, FieldName, tem_val->strValue);
            }
        }
        flag++;
    }
    scan->close(scan);
    return flag;//0
};

#include <parser/statement.h>

/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..] WHERE <logical_expr>
 * update student set username='zw',id='12' where password='123'
 * 解析获得 sql_stmt_update 结构
 */

/*TODO: parse_sql_stmt_update， update语句解析*/
sql_stmt_update *parse_sql_stmt_update(ParserT *parser_zw) {

    ///以update student set sage=20[,fieldname=expr] where sage=22
    char *tableName_zw = NULL;//保存上面update语句的student
    arraylist *fields_zw = arraylist_create();//set fields 被更新的字段列表。保存上面update语句的sage
    arraylist *fieldsExpr_zw = arraylist_create();//set fields expression 新值(表达式)列表 。保存上面update语句的20
    Expression *where = NULL; //保存上面update语句的 sage = 22
    TokenT *token = parseNextToken(parser_zw);  //首先判断当前解析器中的词是否为 NULL,如果是，就获取下一个 token;否则就返回当前词
    if(!matchToken(parser_zw, TOKEN_RESERVED_WORD, "update")){//执行完后，此时token指针已经指向parse的下一个词
        strcpy(parser_zw->parserMessage, "语法错误.");
        return NULL;
    }

    token = parseNextToken(parser_zw);  //首先判断当前解析器中的词是否为 NULL,如果是，就获取下一个 token;否则就返回当前词
    if (token->type == TOKEN_WORD) {
        tableName_zw = new_id_name(); //new_id_name是分配空间，tableName是保存表名
        strcpy(tableName_zw, token->text);
    } else {
        strcpy(parser_zw->parserMessage, "Invalid Sql: Missing Table Name,Error.");
        return NULL;
    }

    token = parseEatAndNextToken(parser_zw);
    if(!matchToken(parser_zw, TOKEN_RESERVED_WORD, "set")){//执行完后，此时token指针已经指向parse的下一个词
        strcpy(parser_zw->parserMessage, "语法错误.");
        return NULL;
    }

    token = parseNextToken(parser_zw);
    while (token != NULL && (token->type == TOKEN_WORD || token->type == TOKEN_COMMA)) {
        if (token->type == TOKEN_COMMA){//*以上是操作符，在表达式解析时使用*/ //逗号
            /*若是逗号，则需要跳过逗号*/
            token = parseEatAndNextToken(parser_zw);
        }
        /*获取字段名*/
        if (token->type == TOKEN_WORD) {//非保留字的词
            char *fieldName = new_id_name();
            strcpy(fieldName, token->text);
            arraylist_add(fields_zw, fieldName);
        } else {
            strcpy(parser_zw->parserMessage, "Invalid Sql: Missing Table Name,Error.");
            return NULL;
        }
        token = parseEatAndNextToken(parser_zw);
        /*字段名后面跟着等号*/
        if (!matchToken(parser_zw, TOKEN_EQ, "=")) {
            strcpy(parser_zw->parserMessage, "Invalid Sql: Missing = .");
            return NULL;
        }
        /*等号后面是表达式*/
        Expression *expr0 = parseExpressionRD(parser_zw);
        arraylist_add(fieldsExpr_zw, expr0);
        token = parseNextToken(parser_zw);
        if (token == NULL) {
            break;
        }
    }

    if (token !=NULL) {
        if (!matchToken(parser_zw, TOKEN_RESERVED_WORD, "where")) {//sql中的保留字，比如create,set,where 等等
            strcpy(parser_zw->parserMessage, "Grammar Error.");
            return NULL;
        }
        where = parseExpressionRD(parser_zw);//已经申请了空间
        if (parser_zw->parserStateType == PARSER_WRONG) {
            return NULL;
        }
    }
    sql_stmt_update *sqlStmtUpdate_zw = (sql_stmt_update *) malloc(sizeof(sql_stmt_update));

    TableReference_t *ref_zw =   TableReference_make(tableName_zw, NULL);//TableReference_make(char *table_name, char *alias)//alias的意思是‘别名’
    SRA_t *table_zw =  SRATable(ref_zw);//创建一个表关系
    sqlStmtUpdate_zw->where = SRASelect(table_zw, where);//SRA_t *SRASelect(SRA_t *sra, Expression *cond)//创建选择关系
    sqlStmtUpdate_zw->tableName = tableName_zw;
    sqlStmtUpdate_zw->fields = fields_zw;
    sqlStmtUpdate_zw->fieldsExpr = fieldsExpr_zw;
    return sqlStmtUpdate_zw;
};
