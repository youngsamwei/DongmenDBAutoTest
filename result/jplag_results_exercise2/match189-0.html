bool is_Match_word_(TokenT* token, TokenType a)
{
    if(token!=NULL && token->type == a)
        return true;
    return  false;
}

bool  parseAssignExpr(ParserT *parser, arraylist *listvalue, arraylist *listExpr) {
    TokenT *token = parseNextToken(parser);
    while (1) {
        // field
        if (token!=NULL && token->type == TOKEN_WORD) {
            //解析field名字
            char *q_a_qq_x_t_y_xty_abcd_fieldName_ts_s_s = new_id_name();
            strcpy(q_a_qq_x_t_y_xty_abcd_fieldName_ts_s_s, token->text);
            arraylist_add(listvalue, q_a_qq_x_t_y_xty_abcd_fieldName_ts_s_s);
            token = parseEatAndNextToken(parser);
            //判断 =
            if (!matchToken(parser, TOKEN_EQ, "=")) {
                return 0;
            }
            // 解析expr的值 expr
            Expression *expr = parseExpressionRD(parser);
            if (parser->parserStateType == PARSER_WRONG) {
                strcpy(parser->parserMessage, "invalid sql: worning!!! worning!worning!worning!worning!worning!worning！ invalid value.");
                return 0;
            }
            arraylist_add(listExpr, expr);
            token = parseNextToken(parser);
            if (token != NULL && token->type == TOKEN_COMMA) { //如果是逗号继续进行
                token = parseEatAndNextToken(parser);
            }else{
                break;
            }
        }
        else {
            strcpy(parser->parserMessage, "worning!!! worning!worning!worning!worning!worning!worning！invalid sql: missing field name.");
            return 0;
        }
    }
    return 1;
};

int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_update 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成update操作
     * */
    size_t Update_xty_res = 0;
    char *x_t_y_sf_tablename_xtyxty = sqlStmtUpdate->tableName;
    arraylist *x_t_y_xty_abcd_updatefields_ts_ss = sqlStmtUpdate->fields;
    arraylist *am_ss_xty_update_Fields_xx_Exprxty_cc = sqlStmtUpdate->fieldsExpr;
    physical_scan *fv_xty_XTY_xTy_update_Scan_llll = physical_scan_generate(db, sqlStmtUpdate->where, tx);
    fv_xty_XTY_xTy_update_Scan_llll->beforeFirst(fv_xty_XTY_xTy_update_Scan_llll);

    while (fv_xty_XTY_xTy_update_Scan_llll->next(fv_xty_XTY_xTy_update_Scan_llll)) {
        for (size_t i = 0; i < x_t_y_xty_abcd_updatefields_ts_ss->size; ++i) {
            char *cntFieldName = arraylist_get(x_t_y_xty_abcd_updatefields_ts_ss, i);
            variant *cntval = (variant *) calloc(sizeof(variant), 1);
            enum data_type field_type = fv_xty_XTY_xTy_update_Scan_llll->getField(fv_xty_XTY_xTy_update_Scan_llll, x_t_y_sf_tablename_xtyxty, cntFieldName)->type;
            physical_scan_evaluate_expression(arraylist_get(am_ss_xty_update_Fields_xx_Exprxty_cc, i), fv_xty_XTY_xTy_update_Scan_llll, cntval);
            if (cntval->type != field_type) {
                fprintf(stdout, "invalid sql: field type mismatched.");
                return DONGMENDB_EINVALIDSQL;
            }
            if (cntval->type == DATA_TYPE_CHAR) {
                /*字符串超出定义时的长度，则截断字符串.*/
                int max_length_str = fv_xty_XTY_xTy_update_Scan_llll->getField(fv_xty_XTY_xTy_update_Scan_llll, x_t_y_sf_tablename_xtyxty, cntFieldName)->length;
                if (max_length_str < strlen(cntval->strValue)) {
                    cntval->strValue[max_length_str] = '\0';
                }
                fv_xty_XTY_xTy_update_Scan_llll->setString(fv_xty_XTY_xTy_update_Scan_llll, x_t_y_sf_tablename_xtyxty, cntFieldName, cntval->strValue);
            }else if (cntval->type == DATA_TYPE_INT) {
                fv_xty_XTY_xTy_update_Scan_llll->setInt(fv_xty_XTY_xTy_update_Scan_llll, x_t_y_sf_tablename_xtyxty, cntFieldName, cntval->intValue);
            }
        }
        Update_xty_res++;
    }
    fv_xty_XTY_xTy_update_Scan_llll->close(fv_xty_XTY_xTy_update_Scan_llll);
    return  Update_xty_res ;;
};
