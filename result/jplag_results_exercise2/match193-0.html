#include <parser/statement.h>

/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..] WHERE <logical_expr>
 * 解析获得 sql_stmt_update 结构
 */

/*TODO: parse_sql_stmt_update， update语句解析*/
sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {

    char *tableName = NULL;//tableName用来保存上面update语句中的表名
    arraylist *fields = arraylist_create();//fields用来保存上面update语句的列名
    arraylist *fieldsExpr = arraylist_create();//fieldExpr用来保存update语句中需要更新的字段
    Expression *where = NULL; //where用来保存update语句中的条件
    //parser解析器作用主要是获取 token(词)，然后对 token(词)进行解析
    TokenT *token = parseNextToken(parser);  //首先判断当前解析器中的词是否为 NULL,如果是，就获取下一个 token;否则就返回当前词
    // matchToken对解析器中 token(词）的TokenType和text进行检查，如果匹配成功，就直接获取下一个词放入解析器中，
    //否则输出错误信息，返回 NUL，不对解析器中的词修改
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "update")){//执行完后，此时token指针已经指向parse的下一个词
        strcpy(parser->parserMessage, "没有匹配到关键词update.");
        return NULL;
    }
    token = parseNextToken(parser);  //首先判断当前解析器中的词是否为 NULL,如果是，就获取下一个 token;否则就返回当前词
    if (token->type == TOKEN_WORD) {
        tableName = new_id_name(); //new_id_name是分配空间，tableName是保存表名
        strcpy(tableName, token->text);
    } else {
        strcpy(parser->parserMessage, "没有匹配到表的名称");
        return NULL;
    }
    token = parseEatAndNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "set")){//执行完后，此时token指针已经指向parse的下一个词
        strcpy(parser->parserMessage, "没有匹配到关键词set.");
        return NULL;
    }
    /*获取字段名*/
    token = parseNextToken(parser);
    while (token != NULL && (token->type == TOKEN_WORD || token->type == TOKEN_COMMA)) {
        if (token->type == TOKEN_COMMA){//*以上是操作符，在表达式解析时使用*/ //逗号
            /*若是逗号，则需要跳过逗号*/
            token = parseEatAndNextToken(parser);
        }
        if (token->type == TOKEN_WORD) {//非保留字的词
            char *fieldName = new_id_name();
            strcpy(fieldName, token->text);
            arraylist_add(fields, fieldName);
        } else {
            strcpy(parser->parserMessage, "没有匹配到表名.");
            return NULL;
        }
        token = parseEatAndNextToken(parser);
        /*字段名后面跟着等号*/
        if (!matchToken(parser, TOKEN_EQ, "=")) {
            strcpy(parser->parserMessage, "没有匹配到关键词 = ");
            return NULL;
        }
        /*等号后面是表达式*/
        Expression *expr0 = parseExpressionRD(parser);
        arraylist_add(fieldsExpr, expr0);
        token = parseNextToken(parser);
        if (token == NULL) {
            break;
        }
    }
    if (token !=NULL) {
        if (!matchToken(parser, TOKEN_RESERVED_WORD, "where")) {//sql中的保留字，比如create,set,where 等等
            strcpy(parser->parserMessage, "没有匹配到关键词where.");
            return NULL;
        }
        where = parseExpressionRD(parser);//已经申请了空间
        if (parser->parserStateType == PARSER_WRONG) {
            return NULL;
        }
    }
    sql_stmt_update *sqlStmtUpdate = (sql_stmt_update *) malloc(sizeof(sql_stmt_update));//

    TableReference_t *ref =   TableReference_make(tableName, NULL);//TableReference_make(char *table_name, char *alias)创建表关系必须用到TableReference关系
    SRA_t *table =  SRATable(ref);//创建一个表关系
    sqlStmtUpdate->where = table;//把表关系放在where里
    if (where != NULL) {
        sqlStmtUpdate->where = SRASelect(table, where);//SRA_t *SRASelect(SRA_t *sra, Expression *cond)//创建选择关系
    }
    sqlStmtUpdate->tableName = tableName;
    sqlStmtUpdate->fields = fields;
    sqlStmtUpdate->fieldsExpr = fieldsExpr;
    return sqlStmtUpdate;
};

int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
//扫描指针
  //  Scan_pointer
    char *Table = sqlStmtUpdate->tableName;
    arraylist *fields = sqlStmtUpdate->fields;
    arraylist *value = sqlStmtUpdate->fieldsExpr;
    physical_scan *Scan_pointer = physical_scan_generate(db, sqlStmtUpdate->where, tx);//创建全表扫描计划
    Scan_pointer->beforeFirst(Scan_pointer);//初始化扫描指针
   int success_num=0;//定义正确执行的条数success_num
    while (Scan_pointer->next(Scan_pointer)) {//当还有扫描计划时,继续扫描
        for (size_t i = 0; i < sqlStmtUpdate->fields->size; ++i) {  //arraylist *fields; //set fields 被更新的字段列表
            char *column_name = arraylist_get(sqlStmtUpdate->fields, i);//获取要更新的列名
            Expression *currentFieldsExprName=arraylist_get(sqlStmtUpdate->fieldsExpr,i);//获取需要更新的值的表达式
            variant *variant1 = (variant *) calloc(sizeof(variant), 1);//保存需要更新的字段类型
            //获取列名的的类型
            enum data_type field_type = Scan_pointer->getField(Scan_pointer, sqlStmtUpdate->tableName, column_name)->type;
            physical_scan_evaluate_expression(currentFieldsExprName, Scan_pointer, variant1);//计算后序表达式的值
//
            if (variant1->type != field_type) {
                fprintf(stdout, "类型不匹配.");
                return -1;//1
            }
            if (variant1->type == DATA_TYPE_INT) {//匹配到int类型
                Scan_pointer->setInt(Scan_pointer, sqlStmtUpdate->tableName, column_name, variant1->intValue);
            } else if (variant1->type == DATA_TYPE_CHAR) {//匹配到char类型
                Scan_pointer->setString(Scan_pointer, sqlStmtUpdate->tableName, column_name, variant1->strValue);
            }
        }
        success_num++;
    }
    Scan_pointer->close(Scan_pointer);//关闭扫描
    return success_num;//返回正确执行的条数
};
