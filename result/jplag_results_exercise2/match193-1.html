#include <parser/statement.h>



sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {

    char *Name = NULL;
    arraylist *ff = arraylist_create();

    arraylist *valuea = arraylist_create();

    Expression *where = NULL;
    TokenT *token = parseNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "update")){
        strcpy(parser->parserMessage, "wrong answer.");
        return NULL;
    }


    token = parseNextToken(parser);

    if (token->type == TOKEN_WORD) {
        Name = new_id_name();
        strcpy(Name, token->text);
    } else {
        strcpy(parser->parserMessage, "wrong: missing table name.");
        return NULL;
    }
    token = parseEatAndNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "set")){
        strcpy(parser->parserMessage, "wrong: missing table set.");
        return NULL;
    }

    token = parseNextToken(parser);
    while (token != NULL && (token->type == TOKEN_WORD || token->type == TOKEN_COMMA)) {
        if (token->type == TOKEN_COMMA){
            token = parseEatAndNextToken(parser);
        }
        if (token->type == TOKEN_WORD) {
            char *fieldName = new_id_name();
            strcpy(fieldName, token->text);
            arraylist_add(ff, fieldName);
        }
        else {
            strcpy(parser->parserMessage, "wrong: missing table name.");
            return NULL;
        }
        token = parseEatAndNextToken(parser);
        if (!matchToken(parser, TOKEN_EQ, "=")) {
            strcpy(parser->parserMessage, "wrong: missing equality.");
            return NULL;
        }
        Expression *expr0 = parseExpressionRD(parser);
        arraylist_add(valuea, expr0);
        token = parseNextToken(parser);
        if (token == NULL) {
            break;

        }
    }


    if (token !=NULL) {
        if (!matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
            strcpy(parser->parserMessage, "wrong: missing where.");
            return NULL;
        }
        where = parseExpressionRD(parser);
        if (parser->parserStateType == PARSER_WRONG) {
            return NULL;
        }
    }
    sql_stmt_update *sqlStmtUpdate = (sql_stmt_update *) malloc(sizeof(sql_stmt_update));

    TableReference_t *ref =   TableReference_make(Name, NULL);
    SRA_t *table =  SRATable(ref);
    sqlStmtUpdate->where = table;
    if (where != NULL) {
        sqlStmtUpdate->where = SRASelect(table, where);
    }
    sqlStmtUpdate->tableName = Name;
    sqlStmtUpdate->fields = ff;
    sqlStmtUpdate->fieldsExpr = valuea;
    return sqlStmtUpdate;
};

#include "physicalplan/physicalplan.h"


int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){

    physical_scan *scan = physical_scan_generate(db, sqlStmtUpdate->where, tx);
    scan->beforeFirst(scan);
    while (scan->next(scan)) {
        for (size_t i = 0; i < sqlStmtUpdate->fields->size; ++i) {
            char *currentFieldName = arraylist_get(sqlStmtUpdate->fields, i);
            Expression *currentFieldsExprName=arraylist_get(sqlStmtUpdate->fieldsExpr,i);
            variant *val = (variant *) calloc(sizeof(variant), 1);
            enum data_type field_type = scan->getField(scan, sqlStmtUpdate->tableName, currentFieldName)->type;
            physical_scan_evaluate_expression(currentFieldsExprName, scan, val);
            if (val->type != field_type) {
                fprintf(stdout, "不合法的sql语句: field type mismatched.");
                return DONGMENDB_EINVALIDSQL;
            }
            if (val->type == DATA_TYPE_INT) {
                scan->setInt(scan, sqlStmtUpdate->tableName, currentFieldName, val->intValue);
            } else if (val->type == DATA_TYPE_CHAR) {
                scan->setString(scan, sqlStmtUpdate->tableName, currentFieldName, val->strValue);
            }
        }
    }
    scan->close(scan);
    return DONGMENDB_OK;
};
