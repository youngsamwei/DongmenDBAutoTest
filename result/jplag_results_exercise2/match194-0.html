#include <parser/statement.h>

/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

/*TODO: parse_sql_stmt_update， update语句解析*/
sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {
    //解析器（parser）作用主要是获取 token(词)，然后对 token(词)进行解析
    char *tableName = NULL;//保存update语句的表名
    arraylist *fields=arraylist_create(); // 被更新的字段列表
    arraylist *fieldsExpr=arraylist_create(); // 新值(表达式)列表
    SRA_t *where = NULL;
    TokenT *token;

    //判断当前Token是否是“update”保留字
    //如果匹配成功，就直接获取下一个词放入解析器中
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {
        return NULL;
    }

    //判断当前解析器中的词是否为 NULL,如果是，就获取下一个 token;否则就返回当前词
    token = parseNextToken(parser);


    //获取表名使用 if 判断是否是 TOKEN_WORD非保留字类型
    if (token && token->type == TOKEN_WORD) {
        tableName = new_id_name();//给字符串指针开空间
        strcpy(tableName, token->text);
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }


	parseEatAndNextToken(parser);//直接获取下一个词


    //判断当前Token是否是“set”
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "set")) {
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }

    token = parseNextToken(parser);
    //循环将更新字段列表和新值表达式列表提取出来
	while (1) {

        if (token == NULL || token->type != TOKEN_WORD) {
            strcpy(parser->parserMessage, "invalid sql: missing field name.");
            return 0;
        }
		//获取filename
        char *fieldName = new_id_name();
        strcpy(fieldName, token->text);
        arraylist_add(fields, fieldName);
        parseEatAndNextToken(parser);

		//判断等号
        if (!matchToken(parser, TOKEN_EQ, "=")) {
            return 0;
        }

		//获取新值表达式，其中会将parser清空
        Expression *expr = parseExpressionRD(parser);
        if (parser->parserStateType == PARSER_WRONG) {
            strcpy(parser->parserMessage, "invalid sql: invalid expression.");
            return 0;
        }
        arraylist_add(fieldsExpr, expr);
		//判断是否为空或者逗号
        token = parseNextToken(parser);
        if (token == NULL || token->type != TOKEN_COMMA) {
            break;
        }
        token = parseEatAndNextToken(parser);
    }


    //获取where表达式


	where = SRATable(TableReference_make(tableName, NULL));
	if (matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
			Expression *whereExpr = parseExpressionRD(parser);
			where = SRASelect(where, whereExpr);
	}


    //对各字段进行赋值，返回sql_stmt_update结构体
     sql_stmt_update *sqlStmtUpdate = (sql_stmt_update *) calloc(sizeof(sql_stmt_update), 1);
    sqlStmtUpdate->tableName = tableName;
    sqlStmtUpdate->fields = fields;
    sqlStmtUpdate->fieldsExpr = fieldsExpr;
    sqlStmtUpdate->where = where;
//    fprintf(stderr, "TODO: update is not implemented yet. in parse_sql_stmt_update \n");

};

#include "physicalplan/physicalplan.h"

/*执行 update 语句*/
/*TODO: plan_execute_update， update语句执行*/
int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_update 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成update操作
     * */

    char *tableName=sqlStmtUpdate->tableName;

    arraylist *fieldNames=sqlStmtUpdate->fields;
    arraylist *values=sqlStmtUpdate->fieldsExpr;

    physical_scan *plan = physical_scan_generate(db, sqlStmtUpdate->where, tx);
    plan->beforeFirst(plan);
	int num  = 0;

    while (plan->next(plan)){
        for (int i = 0; i < fieldNames->size ; i++) {
			char *currentFieldName = arraylist_get(fieldNames, i);
            variant *val = (variant *) calloc(sizeof(variant), 1);
            enum data_type field_type = plan->getField(plan, tableName, currentFieldName)->type;
            physical_scan_evaluate_expression(arraylist_get(values, i), plan, val);

            if (val->type != field_type) {
                fprintf(stdout, "invalid sql: field type mismatched.");
                return -1;
            }

            if (val->type == DATA_TYPE_INT) {
                plan->setInt(plan, tableName, currentFieldName, val->intValue);
            }
			else if (val->type == DATA_TYPE_CHAR) 
			{
                /*字符串超出定义时的长度，则截断字符串.*/
                int max_length_str = plan->getField(plan, tableName, currentFieldName)->length;
                if (max_length_str < strlen(val->strValue)) {
                    val->strValue[max_length_str] = '\0';
                }
                plan->setString(plan, tableName, currentFieldName, val->strValue);
            } 
			
          }

          num++;
        }
    plan->close(plan);
    return num;
};
