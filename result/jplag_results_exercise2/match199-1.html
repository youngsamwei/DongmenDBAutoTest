#include <parser/statement.h>

/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

/*TODO: parse_sql_stmt_update， update语句解析*/

sql_stmt_update *parse_sql_stmt_update(ParserT *parser){
    char *tableName = NULL;
    arraylist *fields = arraylist_create();                                   //更改的变量的值
    arraylist *fieldsExpr = arraylist_create();                                   //更改后的数值

    if (!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {
        return NULL;
    }
    TokenT *token = parseNextToken(parser);   //解析得到操作的表明

    if(token->type == TOKEN_WORD){
        tableName = new_id_name();           //开辟空间
        strcpy(tableName,token->text);
    }else{
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }
    token = parseEatAndNextToken(parser);                      //这个地方要跳过tableName的字段
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "set")) {
        strcpy(parser->parserMessage, "invalid sql: should be set.");
        return NULL;
    }

    token = parseNextToken(parser);

    if (token->type == TOKEN_WORD) {
        while (token->type == TOKEN_WORD) {
            char *fieldName = new_id_name();
            strcpy(fieldName, token->text);
            arraylist_add(fields, fieldName);     //找到更改的列名

            token = parseEatAndNextToken(parser);

            if(token->type == TOKEN_EQ){         //匹配到等号
                token = parseEatAndNextToken(parser);
            } else{
                return NULL;
            }
            Expression *values = parseExpressionRD(parser);
            arraylist_add(fieldsExpr, values);
            token = parseNextToken(parser);

            if (token != NULL && token->type==TOKEN_COMMA){                  //每一个单词后面必须跟一个逗号，除了最后䘝后面为')'之外，如果解析为逗号则吃掉继续
                token = parseEatAndNextToken(parser);
            }else{
                break;
            }

        }
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing set content.");
        return NULL;
    }


    TableReference_t *ref = TableReference_make(tableName,NULL);
    SRA_t * table = SRATable(ref);
    table -> t = SRA_TABLE;

    sql_stmt_update *sqlstmtupdate = (sql_stmt_update *)malloc(sizeof(sql_stmt_update));
    sqlstmtupdate->tableName = tableName;
    sqlstmtupdate->fields = fields;
    sqlstmtupdate->fieldsExpr = fieldsExpr;
    sqlstmtupdate->where = table;

    if(token == NULL)
        return sqlstmtupdate; //没有where的情况直接返回

    token = parseNextToken(parser);
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
        strcpy(parser->parserMessage, "invalid sql: missing where.");
        return NULL;
    }

    //后面匹配where中的内容
    Expression *whereExpr = parseExpressionRD(parser);
    if(parser->parserStateType == PARSER_WRONG){
        return NULL;
    };
    SRA_t *  where = SRASelect(table,whereExpr);                         //Expression
    where->t = SRA_SELECT;

    //准备返回的数据类型
    sqlstmtupdate->where = where;
    return sqlstmtupdate;
};

#include "physicalplan/physicalplan.h"

/*执行 update 语句*/
/*TODO: plan_execute_update， update语句执行*/
int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_update 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成update操作
     * */


    int num = 0;
    //将关系代数表达式转换为扫描计划
    physical_scan *plan = physical_scan_generate(db,sqlStmtUpdate->where,tx);

    arraylist *names = plan->getFieldsName(plan, sqlStmtUpdate->tableName);              //得到整个表的列

    arraylist *_fields = sqlStmtUpdate->fields;

    plan->beforeFirst(plan);
    while (plan->next(plan)) {

        num++;

        for (int name_i = 0; name_i < _fields->size; name_i++) {                        //循环遍历表中的每一列
            char *name = arraylist_get(_fields, name_i);
            variant *var = (variant *)malloc(sizeof(variant) * 1);
            memset(var,0,sizeof(var));
            physical_scan_evaluate_expression( arraylist_get(sqlStmtUpdate->fieldsExpr, name_i),plan,var);
            if (plan->getField(plan, sqlStmtUpdate->tableName, name)->type == DATA_TYPE_CHAR) {
                //长度超出限制判断
                void_ptr *ptr = (void_ptr *) calloc(sizeof(void_ptr *), 1);
                hashmap_get(plan->physicalScanSelect->scan->physicalScanTable->tableInfo->fields, name, ptr);
                field_info *fieldInfo = *ptr;

                //如果长度超出限制，那么截去超出范围的内容
                if(strlen(var->strValue) > fieldInfo->length)
                    var->strValue[fieldInfo->length] = '\0';
                plan->setString(plan, sqlStmtUpdate->tableName, name, var->strValue);
            } else if (plan->getField(plan, sqlStmtUpdate->tableName, name)->type == DATA_TYPE_INT) {
                plan->setInt(plan, sqlStmtUpdate->tableName, name, var->intValue);
            }
        }
    }
    return num;
};
