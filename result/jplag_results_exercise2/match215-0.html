 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

/* parse_sql_stmt_update， update语句解析*/
sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {
    char *w_w_x_tableName = NULL;
    arraylist *fields = arraylist_create();
    arraylist *fieldsExpr = arraylist_create();
    SRA_t *where = NULL;

    TokenT *token;

    // update
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {
        return NULL;
    }

    // table_name
    token = parseNextToken(parser);
    if (token->type == TOKEN_WORD) {
        w_w_x_tableName = new_id_name();
        strcpy(w_w_x_tableName, token->text);
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }

    parseEatAndNextToken(parser);

    // set
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "set")) {
        return NULL;
    }

    // <field1> = <expr1>[, <field2 = <expr2>, ..]
    if (!parseAssignExpr(parser, fields, fieldsExpr)) {
        return NULL;
    }

    // where
    where = SRATable(TableReference_make(w_w_x_tableName, NULL));
    if (matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
        Expression *cond = parseExpressionRD(parser);
        where = SRASelect(where, cond);
    }

    // return
    sql_stmt_update *w_w_x_sqlStmtUpdate = (sql_stmt_update *) calloc(sizeof(sql_stmt_update), 1);
    w_w_x_sqlStmtUpdate->tableName = w_w_x_tableName;
    w_w_x_sqlStmtUpdate->fields = fields;
    w_w_x_sqlStmtUpdate->fieldsExpr = fieldsExpr;
    w_w_x_sqlStmtUpdate->where = where;

    return w_w_x_sqlStmtUpdate;
};

int parseAssignExpr(ParserT *parser, arraylist *fields, arraylist *fieldsExpr) {
    TokenT *token = parseNextToken(parser);
    while (1) {
        // field
        if (token == NULL || token->type != TOKEN_WORD) {
            strcpy(parser->parserMessage, "invalid sql: missing field name.");
            return 0;
        }

        char *w_w_x_fieldName = new_id_name();
        strcpy(w_w_x_fieldName, token->text);
        arraylist_add(fields, w_w_x_fieldName);
        parseEatAndNextToken(parser);

        // =
        if (!matchToken(parser, TOKEN_EQ, "=")) {
            return 0;
        }

        // expr
        Expression *expr = parseExpressionRD(parser);
        if (parser->parserStateType == PARSER_WRONG) {
            strcpy(parser->parserMessage, "invalid sql: invalid expression.");
            return 0;
        }
        arraylist_add(fieldsExpr, expr);

        token = parseNextToken(parser);
        if (token == NULL || token->type != TOKEN_COMMA) {
            break;
        }
        token = parseEatAndNextToken(parser);
    }
    return 1;
};

int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_update 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成update操作
     * */

    //   fprintf(stderr, "TODO: update is not implemented yet. in plan_execute_update \n");

    size_t w_w_x_upcount=0;
    physical_scan *w_w_x_scan=physical_scan_generate(db,sqlStmtUpdate->where,tx);
    w_w_x_scan->beforeFirst(w_w_x_scan);
    while(w_w_x_scan->next(w_w_x_scan)){
        for(size_t i=0;i<sqlStmtUpdate->fieldsExpr->size;++i){
            char *w_w_x_currentFieldName = arraylist_get(sqlStmtUpdate->fields,i);
            variant *w_w_x_val = (variant *)calloc(sizeof(variant),1);
            enum data_type field_type = w_w_x_scan->getField(w_w_x_scan,sqlStmtUpdate->tableName,w_w_x_currentFieldName)->type;

            physical_scan_evaluate_expression(arraylist_get(sqlStmtUpdate->fieldsExpr,i),w_w_x_scan,w_w_x_val);

            if(w_w_x_val->type != field_type){
                fprintf(stdout,"error");
                return DONGMENDB_EINVALIDSQL;
            }
            if(w_w_x_val->type == DATA_TYPE_INT){
                w_w_x_scan->setInt(w_w_x_scan,sqlStmtUpdate->tableName,w_w_x_currentFieldName,w_w_x_val->intValue);
            }else if(w_w_x_val->type == DATA_TYPE_CHAR){
                w_w_x_scan->setString(w_w_x_scan,sqlStmtUpdate->tableName,w_w_x_currentFieldName,w_w_x_val->strValue);
            }
        }
        w_w_x_upcount++;
    }
    w_w_x_scan->close(w_w_x_scan);
    return w_w_x_upcount;
};
