#include <parser/statement.h>

/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

/*TODO: parse_sql_stmt_update， update语句解析*/

sql_stmt_update *parse_sql_stmt_update(ParserT *parser){
   //被更新的字段列表
    arraylist *fields=arraylist_create();
    //新值(表达式)列表
    arraylist *fieldsExpr=arraylist_create();
   //获取token
    TokenT *token = parseNextToken(parser);

    //判断当前Token是否是“update”和保留字,如果不是，返回Null，如果是吃掉当前Token并获取下一Token
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {
        return NULL;
    }

    //将表名称保存下来，若不是表明，则提示错误信息
    token = parseNextToken(parser);
    char *tableName = NULL;
    if (token->type == TOKEN_WORD) {
        tableName = new_id_name();
        strcpy(tableName, token->text);
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }
    //将表明放进tablesExpr中
    SRA_t *tablesExpr = parseTablesExpr(parser);
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }

    //判断当前Token是否是“set”
    token = parseNextToken(parser);
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "set")) {
        strcpy(parser->parserMessage, "invalid sql: should be set.");
        return NULL;
    }
    token = parseNextToken(parser);
    //将更新字段列表和新值表达式列表提取出来
    if (token->type == TOKEN_WORD) {

        while (token->type == TOKEN_WORD) {

            //获取更新字段
            char *fieldName = new_id_name();
            strcpy(fieldName, token->text);
            //将更新字段放进fieds数组中
            arraylist_add(fields, fieldName);
            //吃掉等号
            token = parseEatAndNextToken(parser);
            //获取新值表达式
            token = parseEatAndNextToken(parser);
            //将新值表达式放进fieldExpr数组中
            Expression *fieldExpr = parseExpressionRD(parser);
            arraylist_add(fieldsExpr,fieldExpr);
            //如果是逗号，把逗号吃掉，如果token为NULL或为分号，返回结果（where表达式为空）
            token = parseNextToken(parser);
            if(token == NULL || token->type == TOKEN_SEMICOLON){
                Expression *whereExpr=NULL;
                SRA_t *select = SRASelect(tablesExpr, whereExpr);
                sql_stmt_update *sqlStmtUpdate = (sql_stmt_update *)calloc(sizeof(sql_stmt_update),1);
                sqlStmtUpdate->fields=fields;
                sqlStmtUpdate->fieldsExpr=fieldsExpr;
                sqlStmtUpdate->tableName=tableName;
                sqlStmtUpdate->where=select;
                return sqlStmtUpdate;
            }
           else if (token->type==TOKEN_COMMA){
                token = parseEatAndNextToken(parser);
            }
            else
                break;
        }
    }

    //获取where表达式

    if(!matchToken(parser, TOKEN_RESERVED_WORD, "where")){
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }

    //解析where表达式
    Expression *whereExpr = parseExpressionRD(parser);
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }

    //构建返回值
    SRA_t *select = SRASelect(tablesExpr, whereExpr);
    sql_stmt_update *sqlStmtUpdate = (sql_stmt_update *)calloc(sizeof(sql_stmt_update),1);

    sqlStmtUpdate->fields=fields;
    sqlStmtUpdate->fieldsExpr=fieldsExpr;
    sqlStmtUpdate->tableName=tableName;
    sqlStmtUpdate->where=select;
    return sqlStmtUpdate;

};

int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_update 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成update操作
     * */
     //获取解析后的信息
    char *tableName=sqlStmtUpdate->tableName;
    arraylist *fieldNames=sqlStmtUpdate->fields;
    arraylist *values=sqlStmtUpdate->fieldsExpr;
    SRA_t *sra=sqlStmtUpdate->where;
    int num=0;
    if (sra != NULL) {
        //获取扫描计划
        physical_scan *plan = physical_scan_generate(db,sra,tx);
        plan->beforeFirst(plan);
        //遍历表计划
        while (plan->next(plan)){
            num++;
            //查找每个更新字段
            for (int i = 0; i < fieldNames->size ; i++) {
                variant *var = (variant*)calloc(sizeof(variant),1);
                physical_scan_evaluate_expression(arraylist_get(values,i),plan,var);

                char *fieldName = arraylist_get(fieldNames, i);

                //如果更新值为字符串
                if (var->type == DATA_TYPE_CHAR){
                    /*字符串超出定义时的长度，则截断字符串.*/
                    int max_length=plan->getField(plan,tableName,fieldName)->length;
                    if(max_length<strlen(var->strValue)){
                        var->strValue[max_length] = '\0';
                    }

                    plan->setString(plan, tableName, fieldName, var->strValue);
                }
                //如果是整数
                else if(var->type == DATA_TYPE_INT){
                   plan->setInt(plan, tableName, fieldName, var->intValue);
                }
            }

        }

        printf("\nsuccess.");
        plan->close(plan);
    }
    //printf("%d\n",num);
    return num;
};
