#include <parser/statement.h>

/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

/* parse_sql_stmt_update， update语句解析*/
sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {
    char *cccccssssstableName = NULL;
    arraylist *ccccssssfields = arraylist_create();
    arraylist *ccccssssfieldsExpr = arraylist_create();
    SRA_t *ccccsssswhere = NULL;

    TokenT *ccccsssstoken;

    // update
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {
        return NULL;
    }

    // table_name
    ccccsssstoken = parseNextToken(parser);
    if (ccccsssstoken->type == TOKEN_WORD) {
        cccccssssstableName = new_id_name();
        strcpy(cccccssssstableName, ccccsssstoken->text);
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }
    parseEatAndNextToken(parser);

    // set
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "set")) {
        return NULL;
    }

    // <field1> = <expr1>[, <field2 = <expr2>, ..]
    if (!parseAssignExpr(parser, ccccssssfields, ccccssssfieldsExpr)) {
        return NULL;
    }

    // where
    ccccsssswhere = SRATable(TableReference_make(cccccssssstableName, NULL));
    if (matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
        Expression *ccccsssscond = parseExpressionRD(parser);
        ccccsssswhere = SRASelect(ccccsssswhere, ccccsssscond);
    }

    // return
    sql_stmt_update *sqlStmtUpdate = (sql_stmt_update *) calloc(sizeof(sql_stmt_update), 1);
    sqlStmtUpdate->tableName = cccccssssstableName;
    sqlStmtUpdate->fields = ccccssssfields;
    sqlStmtUpdate->fieldsExpr = ccccssssfieldsExpr;
    sqlStmtUpdate->where = ccccsssswhere;

    return sqlStmtUpdate;
};

int parseAssignExpr(ParserT *parser, arraylist *fields, arraylist *fieldsExpr) {
    TokenT *ccccsssstoken = parseNextToken(parser);
    while (1) {
        // field
        if (ccccsssstoken == NULL || ccccsssstoken->type != TOKEN_WORD) {
            strcpy(parser->parserMessage, "invalid sql: missing field name.");
            return 0;
        }

        char *ccccssssfieldName = new_id_name();
        strcpy(ccccssssfieldName, ccccsssstoken->text);
        arraylist_add(fields, ccccssssfieldName);
        parseEatAndNextToken(parser);

        // =
        if (!matchToken(parser, TOKEN_EQ, "=")) {
            return 0;
        }

        // expr
        Expression *ccccssssexpr = parseExpressionRD(parser);
        if (parser->parserStateType == PARSER_WRONG) {
            strcpy(parser->parserMessage, "invalid sql: invalid expression.");
            return 0;
        }
        arraylist_add(fieldsExpr, ccccssssexpr);

        ccccsssstoken = parseNextToken(parser);
        if (ccccsssstoken == NULL || ccccsssstoken->type != TOKEN_COMMA) {
            break;
        }
        ccccsssstoken = parseEatAndNextToken(parser);
    }
    return 1;
};

int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_update 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成update操作
     * */

    char *ccccsssstableName = sqlStmtUpdate->tableName;
    arraylist *ccccssssfields = sqlStmtUpdate->fields;
    arraylist *ccccssssfieldsExpr = sqlStmtUpdate->fieldsExpr;

    physical_scan *ccccssssscan = physical_scan_generate(db, sqlStmtUpdate->where, tx);
    ccccssssscan->beforeFirst(ccccssssscan);
    size_t updated_lines = 0;

    while (ccccssssscan->next(ccccssssscan)) {
        for (size_t i = 0; i <ccccssssfields->size; ++i) {
            char *ccccsssscurrentFieldName = arraylist_get(ccccssssfields, i);
            variant *ccccssssval = (variant *) calloc(sizeof(variant), 1);
            enum data_type field_type = ccccssssscan->getField(ccccssssscan, ccccsssstableName, ccccsssscurrentFieldName)->type;
            physical_scan_evaluate_expression(arraylist_get(ccccssssfieldsExpr, i), ccccssssscan, ccccssssval);

            if (ccccssssval->type != field_type) {
                fprintf(stdout, "invalid sql: field type mismatched.");
                return -1;
            }

            if (ccccssssval->type == DATA_TYPE_INT) {
                ccccssssscan->setInt(ccccssssscan, ccccsssstableName, ccccsssscurrentFieldName, ccccssssval->intValue);
            } else if (ccccssssval->type == DATA_TYPE_CHAR) {
                /*字符串超出定义时的长度，则截断字符串.*/
                int max_length_str = ccccssssscan->getField(ccccssssscan, ccccsssstableName, ccccsssscurrentFieldName)->length;
                if (max_length_str < strlen(ccccssssval->strValue)) {
                    ccccssssval->strValue[max_length_str] = '\0';
                }
                ccccssssscan->setString(ccccssssscan, ccccsssstableName, ccccsssscurrentFieldName, ccccssssval->strValue);
            } else {
                // other data type...
            }
        }
        updated_lines += 1;
    }
    ccccssssscan->close(ccccssssscan);
    fprintf(stdout, " updated %d lines.\n", updated_lines);

    return updated_lines;
};
