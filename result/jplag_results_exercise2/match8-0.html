#include <parser/statement.h>

/**
 * 在现有实现基础上，实现update from子句
 *
 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */
int parseAssignExpr(ParserT *parser, arraylist *fields, arraylist *fieldsExpr);

/* parse_sql_stmt_update， update语句解析*/
sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {
    char *TN = NULL;
    arraylist *FS = arraylist_create();
    arraylist *FE = arraylist_create();
    SRA_t *We = NULL;

    TokenT *token;


    if (!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {
        return NULL;
    }


    token = parseNextToken(parser);
    if (token && token->type == TOKEN_WORD) {
        TN = new_id_name();
        strcpy(TN, token->text);
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }

    parseEatAndNextToken(parser);


    if (!matchToken(parser, TOKEN_RESERVED_WORD, "set")) {
        return NULL;
    }


    if (!parseAssignExpr(parser, FS, FE)) {
        return NULL;
    }


    We = SRATable(TableReference_make(TN, NULL));
    if (matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
        Expression *cond = parseExpressionRD(parser);
        We = SRASelect(We, cond);
    }


    sql_stmt_update *sqlStmtUpdate = (sql_stmt_update *) calloc(sizeof(sql_stmt_update), 1);
    sqlStmtUpdate->tableName = TN;
    sqlStmtUpdate->fields = FS;
    sqlStmtUpdate->fieldsExpr = FE;
    sqlStmtUpdate->where = We;

    return sqlStmtUpdate;
};

int parseAssignExpr(ParserT *parser, arraylist *fields, arraylist *fieldsExpr) {
    TokenT *token = parseNextToken(parser);
    while (1) {

        if (token == NULL || token->type != TOKEN_WORD) {
            strcpy(parser->parserMessage, "invalid sql: missing field name.");
            return 0;
        }

        char *fn = new_id_name();
        strcpy(fn, token->text);
        arraylist_add(fields, fn);
        parseEatAndNextToken(parser);


        if (!matchToken(parser, TOKEN_EQ, "=")) {
            return 0;
        }


        Expression *EX = parseExpressionRD(parser);
        if (parser->parserStateType == PARSER_WRONG) {
            strcpy(parser->parserMessage, "invalid sql: invalid expression.");
            return 0;
        }
        arraylist_add(fieldsExpr, EX);

        token = parseNextToken(parser);
        if (token == NULL || token->type != TOKEN_COMMA) {
            break;
        }
        token = parseEatAndNextToken(parser);
    }
    return 1;
};

#include "physicalplan/physicalplan.h"

/*执行 update 语句的物理计划，返回修改的记录条数
 * 返回大于等于0的值，表示修改的记录条数；
 * 返回小于0的值，表示修改过程中出现错误。
 * */

int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){

    char *TN = sqlStmtUpdate->tableName;
    arraylist *FS = sqlStmtUpdate->fields;
    arraylist *FE = sqlStmtUpdate->fieldsExpr;

    physical_scan *scan = physical_scan_generate(db, sqlStmtUpdate->where, tx);
    scan->beforeFirst(scan);
    size_t updated_lines = 0;

    while (scan->next(scan)) {
        for (size_t i = 0; i < FS->size; ++i) {
            char *currentFieldNameeeeeeee = arraylist_get(FS, i);
            variant *val = (variant *) calloc(sizeof(variant), 1);
            enum data_type field_type = scan->getField(scan, TN, currentFieldNameeeeeeee)->type;
            physical_scan_evaluate_expression(arraylist_get(FE, i), scan, val);

            if (val->type != field_type) {
                fprintf(stdout, "invalid sql: field type mismatched.");
                return -1;
            }

            if (val->type == DATA_TYPE_INT) {
                scan->setInt(scan, TN, currentFieldNameeeeeeee, val->intValue);
            } else if (val->type == DATA_TYPE_CHAR) {

                int max_length_strrrrr = scan->getField(scan, TN, currentFieldNameeeeeeee)->length;
                if (max_length_strrrrr < strlen(val->strValue)) {
                    val->strValue[max_length_strrrrr] = '\0';
                }
                scan->setString(scan, TN, currentFieldNameeeeeeee, val->strValue);
            } else {

            }
        }
        updated_lines += 1;
    }
    scan->close(scan);
    fprintf(stdout, " updated %d lines.\n", updated_lines);

    return updated_lines;
};
