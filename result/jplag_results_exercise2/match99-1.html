 * 支持的update语法：
 *
 * UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_update 结构
 */

sql_stmt_update *parse_sql_stmt_update(ParserT *parser) {
    char *biaoming = NULL;
    arraylist *asdf = arraylist_create();
    arraylist *asdfg = arraylist_create();
    SRA_t *wh = NULL;
    TokenT *to;
    // 首先匹配 update, 使用matchToken(parser,TOKEN_RESERVED_WORD,"update")匹配
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "update")) {
        return NULL;
    }
    // 获取表名 使用 if 判断是否是 TOKEN_WORD 类型，如果是，获得表名；给字符串指针开空间的时候可以使用new_id_name()函数
    to = parseNextToken(parser);
    if (to && to->type == TOKEN_WORD) {
        biaoming = new_id_name();
        strcpy(biaoming, to->text);
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }
    parseEatAndNextToken(parser);
    // 匹配 set 使用 matchToken(parser,TOKEN_RESERVED_WORD,"set")
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "set")) {
        return NULL;
    }
    // 使用循环获得字段名（TOKEN_WORD 类型），放入一个 fields(arraylist 类型)中；循环获得值（或者表达式）parseExpressionRD(parse)函数,fieldsExpr(arraylist 类型)中
    if (!parseAssignExpr(parser, asdf, asdfg)) {
        return NULL;
    }
    // 匹配 where，matchToken(parser,TOKEN_RESERVED_WORD,"where")，注意不是所有update语句都伴随一个where表达式
    wh = SRATable(TableReference_make(biaoming, NULL));
    if (matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
        Expression *cond = parseExpressionRD(parser);//使用parseExpressionRD(parse)获取值（或者表达式），并包裹在SRA_t中（使用SRA_SELECT构造），作为sql_stmt_update->where的值。
        wh = SRASelect(wh, cond);
    }
    // 创建sql_stmt_update指针，开空间，对各字段进行赋值，返回sql_stmt_update结构体
    sql_stmt_update *sqlStmtUpdate = (sql_stmt_update *) calloc(sizeof(sql_stmt_update), 1);
    sqlStmtUpdate->tableName = biaoming;
    sqlStmtUpdate->fields = asdf;
    sqlStmtUpdate->fieldsExpr = asdfg;
    sqlStmtUpdate->where = wh;
    return sqlStmtUpdate;
};

int parseAssignExpr(ParserT *parser, arraylist *fields, arraylist *fieldsExpr) {
    TokenT *token = parseNextToken(parser);
    while (1) {
        if (token == NULL || token->type != TOKEN_WORD) {
            strcpy(parser->parserMessage, "invalid sql: missing field name.");
            return 0;
        }
        char *fieldName = new_id_name();
        strcpy(fieldName, token->text);
        arraylist_add(fields, fieldName);
        parseEatAndNextToken(parser);
        if (!matchToken(parser, TOKEN_EQ, "=")) {
            return 0;
        }
        Expression *expr = parseExpressionRD(parser);
        if (parser->parserStateType == PARSER_WRONG) {
            strcpy(parser->parserMessage, "invalid sql: invalid expression.");
            return 0;
        }
        arraylist_add(fieldsExpr, expr);
        token = parseNextToken(parser);
        if (token == NULL || token->type != TOKEN_COMMA) {
            break;
        }
        token = parseEatAndNextToken(parser);
    }
    return 1;
};

int plan_execute_update(dongmendb *db, sql_stmt_update *sqlStmtUpdate , transaction *tx){
/*删除语句以select的物理操作为基础实现。
* 1. 使用 sql_stmt_update 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
* 2. 执行 select 的物理计划，完成update操作
* */
    char *biaoming = sqlStmtUpdate->tableName;
    arraylist *asd = sqlStmtUpdate->fields;
    arraylist *asdf = sqlStmtUpdate->fieldsExpr;
    physical_scan *scan = physical_scan_generate(db, sqlStmtUpdate->where, tx);
    scan->beforeFirst(scan);
    arraylist *fi;
    size_t updated_lines = 0;
    //使用 where 条件的 SRA_t 构造执行计划遍历计划（循环）
    while (scan->next(scan)) {
        //在遍历计划过程的循环中，计算表达式的值，计算需要修改字段的偏移量
        for (size_t i = 0; i < asd->size; ++i) {
            char *currentFieldName = arraylist_get(asd, i);
            variant *val = (variant *) calloc(sizeof(variant), 1);
            enum data_type field_type = scan->getField(scan, biaoming, currentFieldName)->type;
            physical_scan_evaluate_expression(arraylist_get(asdf, i), scan, val);
            if (val->type != field_type) {
                fprintf(stdout, "invalid sql: field type mismatched.");
                return -1;
            }
            if (val->type == DATA_TYPE_INT) {
                scan->setInt(scan, biaoming, currentFieldName, val->intValue);
            } else if (val->type == DATA_TYPE_CHAR) {
                int max_length_str = scan->getField(scan, biaoming, currentFieldName)->length;
                if (max_length_str < strlen(val->strValue)) {
                    val->strValue[max_length_str] = '\0';
                }
                scan->setString(scan, biaoming, currentFieldName, val->strValue);
            } else {
            }
        }
        //在循环中更新修改的记录条数，作为最终的返回值。
        updated_lines += 1;
    }
    scan->close(scan);
    fprintf(stdout, " updated %d lines.\n", updated_lines);
    return updated_lines;
};
