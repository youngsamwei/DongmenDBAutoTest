#include "physicalplan/physicalplan.h"

/*执行delete语句*/


int plan_execute_delete(dongmendb *db, sql_stmt_delete *sqlStmtDelete,  transaction *tx){

    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_delete 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成 delete 操作
     * */

    //获取where表达式
    SRA_t *where = sqlStmtDelete->where;
    int count = -1;
    //当表达式不为空时创建物理计划
    if(where != NULL)
    {
        count = 0;
        //创建物理计划
        physical_scan *plan = plan_execute_select(db, where, tx);

        plan->beforeFirst(plan);
        while(plan->next(plan))
        {
            //执行删除操作
            plan->deleterec(plan);
            count++;
        }
        printf("\nsuccess.");
        plan->close(plan);
    }

    return count;
};

#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */

sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){

    TokenT *token = parseNextToken(parser);

    //解析是否为delete关键字 否则报错
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "delete"))
    {
        strcpy(parser->parserMessage, "invalid sql: missing keyword DELETE.");
        return  NULL;
    }

    token = parseNextToken(parser);
    if(token->type == TOKEN_MULTIPLY)
    {
        token = parseEatAndNextToken(parser);

    }
    //解析是否为form关键字 进而越过from关键字
    token = parseNextToken(parser);
    matchToken(parser, TOKEN_RESERVED_WORD, "from");

    //解析tableName
    token = parseNextToken(parser);
    char *tableName = NULL;
    if (token->type == TOKEN_WORD)
    {
        tableName = new_id_name();
        strcpy(tableName, token->text);
    }
    else
    {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }

    //封装SRASelect
    TableReference_t *ref = TableReference_make(tableName, NULL);
    SRA_t *tablesExpr = SRATable(ref);

    //创建sql_stmt_delete指针 s并赋值
    sql_stmt_delete *sqlStmtDelete = (sql_stmt_delete *)malloc(sizeof(sql_stmt_delete));

    // 为变量复制
    sqlStmtDelete->tableName = tableName;
    sqlStmtDelete->where = tablesExpr;
    token = parseNextToken(parser);

    //检查语句是否结束

    token = parseEatAndNextToken(parser);
    if(token == NULL || token->type == TOKEN_SEMICOLON)
    {
        return sqlStmtDelete;
    }

    //解析是否为where关键字 否则报错
    token = parseNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "where"))
    {
        strcpy(parser->parserMessage, "invalid sql: missing keyword, WHERE");
        return NULL;
    }

    //解析where后表达式
    token = parseNextToken(parser);
    Expression *whereExpr  = parseExpressionRD(parser);
    if(parser->parserStateType == PARSER_WRONG)
    {
        return NULL;
    }

    //封装SRA
    SRA_t *select = SRASelect(tablesExpr, whereExpr);
    sqlStmtDelete->where = select;


    //返回sql_stmt_update指针
    return sqlStmtDelete;

};
