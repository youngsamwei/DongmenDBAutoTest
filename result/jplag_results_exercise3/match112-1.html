#include <parser/statement.h>



sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser_sql){
    char *DataManager = NULL;
    SRA_t *Context_sra = NULL;
    TokenT *token_word;
    arraylist *scope = arraylist_create();
    arraylist *scopeExpr = arraylist_create();

    if (!matchToken(parser_sql, TOKEN_RESERVED_WORD, "delete") ) {
        return NULL;
    }
    token_word = parseNextToken(parser_sql);
    if (token_word && token_word->type == TOKEN_WORD) {
        DataManager = new_id_name();
        strcpy(DataManager, token_word->text);
    } else {
        strcpy(parser_sql->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }
    parseEatAndNextToken(parser_sql);
    if (!matchToken(parser_sql, TOKEN_RESERVED_WORD, "where")) {
        return NULL;
    }
    SRA_t *sra_table = SRATable(TableReference_make(DataManager, NULL));
    Expression *expre_pre = parseExpressionRD(parser_sql);
    Context_sra = SRASelect(sra_table, expre_pre);
    sql_stmt_delete *database_deleted = (sql_stmt_delete *) calloc(sizeof(sql_stmt_delete), 1);
    database_deleted->tableName = DataManager;
    database_deleted->where = Context_sra;
    return database_deleted;
};
