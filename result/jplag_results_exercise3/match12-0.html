int plan_execute_delete(dongmendb *db, sql_stmt_delete *sqlStmtDelete,  transaction *tx){

    /*删除语句以select的物理操作为基础实现。
     * 1. 使用 sql_stmt_delete 的条件参数，调用 physical_scan_select_create 创建select的物理计划并初始化;
     * 2. 执行 select 的物理计划，完成 delete 操作
     * */

    char *tableName = sqlStmtDelete->tableName;

    physical_scan *saomiao = physical_scan_generate(db, sqlStmtDelete->where, tx);
    saomiao->beforeFirst(saomiao);
    size_t deleted_lines = 0;

    while (saomiao->next(saomiao)) {
        saomiao->deleterec(saomiao);

        deleted_lines += 1;
    }
    saomiao->close(saomiao);
    fprintf(stdout, " updated %d lines.\n", deleted_lines);

    return deleted_lines;
};

#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *

 * 解析获得 sql_stmt_delete 结构
 */

sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){
    char *BM = NULL;
    SRA_t *WHHHH = NULL;
    TokenT *BOOOO;
    arraylist *fields = arraylist_create();
    arraylist *fieldsExpr = arraylist_create();
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "delete") ) {
        return NULL;
    }
    BOOOO = parseNextToken(parser);
    if (BOOOO && BOOOO->type == TOKEN_WORD) {
        BM = new_id_name();
        strcpy(BM, BOOOO->text);
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }
    parseEatAndNextToken(parser);
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
        return NULL;
    }
    SRA_t *table = SRATable(TableReference_make(BM, NULL));
    Expression *cond = parseExpressionRD(parser);
    WHHHH = SRASelect(table, cond);
    sql_stmt_delete *sqlStmtDelete = (sql_stmt_delete *) calloc(sizeof(sql_stmt_delete), 1);
    sqlStmtDelete->tableName = BM;
    sqlStmtDelete->where = WHHHH;
    return sqlStmtDelete;
};
