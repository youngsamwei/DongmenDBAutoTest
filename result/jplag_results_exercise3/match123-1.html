#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */
sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){
    TokenT *token = parseNextToken(parser);
    char *tablename=NULL;
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "delete")) //判断delete
    {
        return NULL;
    }
    token = parseNextToken(parser);

    if(!matchToken(parser, TOKEN_RESERVED_WORD, "from")){
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }
    token = parseNextToken(parser);
    if(token->type==TOKEN_WORD)//获取表名 使用 if 判断是否是 TOKEN_WORD 类型，如果是，获得表名；给字符串指针开空间的时候可以使用new_id_name()函数若不是表名，则提示错误信息
    {
        tablename=new_id_name();
        strcpy(tablename,token->text);
    }else{
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }

    //对有无where的两种情况进行区分讨论
    sql_stmt_delete *sqlStmtDelete=(sql_stmt_delete*)malloc(sizeof(sql_stmt_delete));
    sqlStmtDelete->tableName=tablename;
    TableReference_t *xj=TableReference_make(tablename,NULL);
    SRA_t*tablexpr=SRATable(xj);
    token = parseEatAndNextToken(parser);
    if(token == NULL || token->type == TOKEN_SEMICOLON)//匹配 where，matchToken(parser,TOKEN_RESERVED_WORD,"where")，注意不是所有delete语句都伴随一个where表达式
    {
        sqlStmtDelete->where=tablexpr;
        return sqlStmtDelete;
    }else{
        if (!matchToken(parser, TOKEN_RESERVED_WORD, "where"))//where出错丢失时
        {
            return NULL;
        }
        token=parseNextToken(parser);
        Expression*wherexpr=parseExpressionRD(parser);//使用parseExpressionRD(parse)获取值（或者表达式），并包裹在SRA_t中（使用SRA_SELECT构造）. 作为sql_stmt_delete->where的值。
        if(parser->parserStateType==PARSER_WRONG){
            return NULL;
        }
        SRA_t *where=SRASelect(tablexpr,wherexpr);
        sqlStmtDelete->where=where;//创建sql_stmt_delete指针，开空间，对各字段进行赋值，返回sql_stmt_delete结构体
        return sqlStmtDelete;
    }
};
