#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */
char * parseTableName(ParserT *parser) {
    TokenT *Token = parseNextToken(parser);
    if (Token->type != TOKEN_WORD) {
        strcpy(parser->parserMessage, "语法错误，请检查输入.");
        return NULL;
    } else {
        char *Table_name = new_id_name();
        strcpy(Table_name, Token->text);
        parseEatAndNextToken(parser);
        return Table_name;
    }

}
sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){
    TokenT * Token;
    sql_stmt_delete *Sql_stmt_delete = (sql_stmt_delete *)calloc(sizeof(sql_stmt_delete),1);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "delete")) {
        strcpy(parser->parserMessage, "语法错误，请检查输入");
        return NULL;
    }

    char * Table_name = parseTableName(parser);
    Sql_stmt_delete->tableName = Table_name;
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    Token = parseNextToken(parser);
    if (Token == NULL || Token->type == TOKEN_SEMICOLON) {
        Sql_stmt_delete->where = SRATable(TableReference_make(Table_name, NULL));
        return Sql_stmt_delete;
    }
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "where")){
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }
    Expression *Where_expr = parseExpressionRD(parser);
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    Token = parseNextToken(parser);
    if (Token == NULL || Token->type == TOKEN_SEMICOLON) {
        TableReference_t *Ref = TableReference_make(Table_name, NULL);
        SRA_t *Table =  SRATable(Ref);
        SRA_t *Select = SRASelect(Table, Where_expr);
        Sql_stmt_delete->where = Select;
        return Sql_stmt_delete;
    }
};
