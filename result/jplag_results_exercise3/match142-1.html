#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */
char * parseTableName(ParserT *parser) {
    TokenT *token = parseNextToken(parser);
    if (token->type == TOKEN_WORD) {
        char *w_w_x_tableName = new_id_name();
        strcpy(w_w_x_tableName, token->text);
        parseEatAndNextToken(parser);
        return w_w_x_tableName;
    } else {
        strcpy(parser->parserMessage, "语法错误，请检查输入.");
        return NULL;

    }

}
sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){
    TokenT * w_w_x_token;
    sql_stmt_delete *w_w_x_sqlStmtDelete = (sql_stmt_delete *)calloc(sizeof(sql_stmt_delete),1);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "delete")) {
        strcpy(parser->parserMessage, "语法错误，请检查输入");
        return NULL;
    }

    char * w_w_x_tableName = parseTableName(parser);
    w_w_x_sqlStmtDelete->tableName = w_w_x_tableName;
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    w_w_x_token = parseNextToken(parser);
    if (w_w_x_token == NULL || w_w_x_token->type == TOKEN_SEMICOLON) {
        w_w_x_sqlStmtDelete->where = SRATable(TableReference_make(w_w_x_tableName, NULL));
        return w_w_x_sqlStmtDelete;
    }
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "where")){
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }
    Expression *whereExpr = parseExpressionRD(parser);
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    w_w_x_token = parseNextToken(parser);
    if (w_w_x_token == NULL || w_w_x_token->type == TOKEN_SEMICOLON) {
        TableReference_t *ref = TableReference_make(w_w_x_tableName, NULL);
        SRA_t *table =  SRATable(ref);
        SRA_t *select = SRASelect(table, whereExpr);
        w_w_x_sqlStmtDelete->where = select;
        return w_w_x_sqlStmtDelete;
    }
};
