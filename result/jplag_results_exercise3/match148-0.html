#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */

sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser_zw){
    char *tableName = NULL;//保存上面delete语句的student
    Expression *where_zw = NULL; //保存上面delete语句的 sage = 22
    TokenT *token_zw = parseNextToken(parser_zw);  //首先判断当前解析器中的词是否为 NULL,如果是，就获取下一个 token;否则就返回当前词
    if(!matchToken(parser_zw, TOKEN_RESERVED_WORD, "delete")){//执行完后，此时token指针已经指向parse的下一个词
        strcpy(parser_zw->parserMessage, "Grammar Error.");
        return NULL;
    }

    token_zw = parseNextToken(parser_zw);  //首先判断当前解析器中的词是否为 NULL,如果是，就获取下一个 token;否则就返回当前词
    if (token_zw->type == TOKEN_WORD) {
        tableName = new_id_name(); //new_id_name是分配空间，tableName是保存表名
        strcpy(tableName, token_zw->text);
    } else {
        strcpy(parser_zw->parserMessage, "Invalid Sql: Missing Table Name,Error.");
        return NULL;
    }

    token_zw = parseEatAndNextToken(parser_zw);
    if (token_zw !=NULL) {
        if (!matchToken(parser_zw, TOKEN_RESERVED_WORD, "where")) {//sql中的保留字，比如create,set,where 等等
            strcpy(parser_zw->parserMessage, "Grammar Error.");
            return NULL;
        }
        where_zw = parseExpressionRD(parser_zw);//已经申请了空间
        if (parser_zw->parserStateType == PARSER_WRONG) {
            return NULL;
        }
    }

    sql_stmt_delete *sqlStmtDelete_test = (sql_stmt_delete *) malloc(sizeof(sql_stmt_delete));

    TableReference_t *ref_test =   TableReference_make(tableName, NULL);//TableReference_make(char *table_name, char *alias)//alias的意思是‘别名’
    SRA_t *table_test =  SRATable(ref_test);//创建一个表关系
    sqlStmtDelete_test->where = SRASelect(table_test, where_zw);//SRA_t *SRASelect(SRA_t *sra, Expression *cond)//创建选择关系
    sqlStmtDelete_test->tableName = tableName;

    return sqlStmtDelete_test;

};
