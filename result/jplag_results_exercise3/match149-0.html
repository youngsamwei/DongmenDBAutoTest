/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */
bool IS_MATCH_WTF(TokenT* token, TokenType a);

sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){
    SRA_t *uu_yy_x_t_y_ss_where_xtyxty = NULL;
    TokenT *nns_res_xty_xxty_up_token_live_of = NULL;
    char *q_a_qq_x_t_y_xty_abcd_tableName_ts_s_s = NULL;
    // update <table> set <> = <>
    //判断delete是否存在
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "delete")) {
        strcpy(parser->parserMessage, "worning!!! worning!!!worning!!!invalid sql: delete.");
        return NULL;
    }
    //判断表是否存在
    nns_res_xty_xxty_up_token_live_of = parseNextToken(parser);
    //  fprintf(stderr,token->text);
    if (IS_MATCH_WTF(nns_res_xty_xxty_up_token_live_of,TOKEN_WORD)) {
        q_a_qq_x_t_y_xty_abcd_tableName_ts_s_s = new_id_name();
        strcpy(q_a_qq_x_t_y_xty_abcd_tableName_ts_s_s, nns_res_xty_xxty_up_token_live_of->text);
    } else {
        strcpy(parser->parserMessage, "worning!!! worning!worning!worning!worning!worning!worning！invalid sql: missing table name.");
        return NULL;
    }
    //where
    //表达式
    uu_yy_x_t_y_ss_where_xtyxty = SRATable(TableReference_make(q_a_qq_x_t_y_xty_abcd_tableName_ts_s_s, NULL));
    nns_res_xty_xxty_up_token_live_of = parseEatAndNextToken(parser);//Token(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
        strcpy(parser->parserMessage, "worning!!! worning!worning!worning!worning!worning!worning！invalid sql: missing Where.");
        return NULL;
    } else
    {
        Expression *whereExpr = parseExpressionRD(parser);
        uu_yy_x_t_y_ss_where_xtyxty = SRASelect(uu_yy_x_t_y_ss_where_xtyxty, whereExpr);
    }
    sql_stmt_delete  *sqlStmtDelete = (sql_stmt_delete *)calloc(sizeof(sql_stmt_delete),1);
    sqlStmtDelete->tableName = q_a_qq_x_t_y_xty_abcd_tableName_ts_s_s;
    sqlStmtDelete->where = uu_yy_x_t_y_ss_where_xtyxty;
    // fprintf(stderr,"worning!!! worning!worning!worning!worning!worning!worning！the end happying ending......\n");
    return sqlStmtDelete;
};

bool IS_MATCH_WTF(TokenT* token, TokenType a)
{
    if(token!=NULL && token->type == a)
        return true;
    return  false;
}
