#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */

sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){
    char * biaoming =NULL;
    sql_stmt_delete *anewtable = (sql_stmt_delete *)calloc(sizeof(sql_stmt_delete),1);
    TokenT *TTTT_exchange = parseNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "delete")) {
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }
    /*  if(!matchToken(parser, TOKEN_RESERVED_WORD, "from")) {
          strcpy(parser->parserMessage, "语法错误.");
          return NULL;
       }*/
    TTTT_exchange = parseNextToken(parser);

    if (TTTT_exchange!=NULL && TTTT_exchange->type == TOKEN_WORD) {
        biaoming = new_id_name();
        strcpy(biaoming, TTTT_exchange->text);
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }
    anewtable->tableName = biaoming;
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    TTTT_exchange = parseNextToken(parser);
    if (TTTT_exchange == NULL || TTTT_exchange->type == TOKEN_SEMICOLON) {
        TableReference_t *refance =   TableReference_make(biaoming, NULL);
        SRA_t *biaoaa =  SRATable(refance);
        anewtable->where = biaoaa;
        return anewtable;
    }
    TTTT_exchange = parseEatAndNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
        strcpy(parser->parserMessage, "error where.");
        return NULL;
    }
    TTTT_exchange = parseNextToken(parser);
    Expression *wEr = parseExpressionRD(parser);
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    TTTT_exchange = parseNextToken(parser);
    if (TTTT_exchange == NULL || TTTT_exchange->type == TOKEN_SEMICOLON) {
        TableReference_t *refance =   TableReference_make(biaoming, NULL);
        SRA_t *biaoaa =  SRATable(refance);
        SRA_t *slt = SRASelect(biaoaa, wEr);
        anewtable->where = slt;
        return anewtable;
    }
};
