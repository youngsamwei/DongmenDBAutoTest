#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */

sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){

    char *TName = NULL;

    //匹配保留字delete
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "delete")){
        strcpy(parser->parserMessage, "Error: delete should be into.please add it.");
        return NULL;
    }

    TokenT *token = parseNextToken(parser);

    if (token->type == TOKEN_WORD) {
        //开辟空间，保存表名
        TName = new_id_name();
        strcpy(TName, token->text);
    } else {
        strcpy(parser->parserMessage, "Error: missing table name.please add it.");
        return NULL;
    }
    token = parseEatAndNextToken(parser);

    //构造where表达式
    Expression *WExpr = NULL;
    if (token !=NULL) {
        if (!matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
            strcpy(parser->parserMessage,"Error: should be where.please add it.");
            return NULL;
        }
        WExpr = parseExpressionRD(parser);
        if (parser->parserStateType == PARSER_WRONG) {
            return NULL;
        }
    }
    sql_stmt_delete *SqlStmtDelete=(sql_stmt_delete*)malloc(sizeof(sql_stmt_delete));
    SqlStmtDelete->tableName = TName;

    TableReference_t *al=   TableReference_make(SqlStmtDelete->tableName, NULL);
    SRA_t *Tzl =  SRATable(al);
    SqlStmtDelete->where = Tzl;
    if (WExpr != NULL) {
        SqlStmtDelete->where = SRASelect(Tzl, WExpr);
    }
    return SqlStmtDelete;
};
