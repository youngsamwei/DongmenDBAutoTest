#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */

char * parseTableName(ParserT *p) {
    TokenT *token = parseNextToken(p);
    if (token->type == TOKEN_WORD) {
        char *tableName = new_id_name();
        strcpy(tableName, token->text);
        parseEatAndNextToken(p);
        return tableName;
    } else {
        strcpy(p->parserMessage, "语法错误，请检查输入.");
        return NULL;
    }

}
sql_stmt_delete *parse_sql_stmt_delete(ParserT *p){
    TokenT * token;
    sql_stmt_delete *sqlStmtDelete = (sql_stmt_delete *)calloc(sizeof(sql_stmt_delete),1);
    if(!matchToken(p, TOKEN_RESERVED_WORD, "delete")) {
        strcpy(p->parserMessage, "语法错误，请检查输入");
        return NULL;
    }

    if(!matchToken(p, TOKEN_RESERVED_WORD, "from")) {
        strcpy(p->parserMessage, "语法错误，请检查输入.");
        return NULL;
    }
    char * tableName = parseTableName(p);
    sqlStmtDelete->tableName = tableName;
    if (p->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    token = parseNextToken(p);
    if (token == NULL || token->type == TOKEN_SEMICOLON) {
        sqlStmtDelete->where = SRATable(TableReference_make(tableName, NULL));
        return sqlStmtDelete;
    }
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "where")){
        strcpy(p->parserMessage, "语法错误.");
        return NULL;
    }
    Expression *whereExpr = parseExpressionRD(p);
    if (p->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    token = parseNextToken(p);
    if (token == NULL || token->type == TOKEN_SEMICOLON) {
        TableReference_t *r = TableReference_make(tableName, NULL);
        SRA_t *table =  SRATable(r);
        SRA_t *select = SRASelect(table, whereExpr);
        sqlStmtDelete->where = select;
        return sqlStmtDelete;
    }
};
