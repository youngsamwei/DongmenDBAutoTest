#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */
char * parseTableName(ParserT *p_wzs_T) {
    TokenT *T_wzs_T = parseNextToken(p_wzs_T);
    if (T_wzs_T ->type != TOKEN_WORD) {
        strcpy(p_wzs_T->parserMessage, "语法错误，请检查输入.");
        return NULL;
    } else {
        char *name_of_wzs_table = new_id_name();
        strcpy(name_of_wzs_table, T_wzs_T ->text);
        parseEatAndNextToken(p_wzs_T);
        return name_of_wzs_table;
    }

}
sql_stmt_delete *parse_sql_stmt_delete(ParserT *p_wzs_T){
    TokenT * T_wzs_T;
    sql_stmt_delete *S_wzs_d = (sql_stmt_delete *)calloc(sizeof(sql_stmt_delete),1);
    if(!matchToken(p_wzs_T, TOKEN_RESERVED_WORD, "delete")) {
        strcpy(p_wzs_T->parserMessage, "语法错误，请检查输入");
        return NULL;
    }

    char * name_of_wzs_table = parseTableName(p_wzs_T);
    S_wzs_d->tableName = name_of_wzs_table;
    if (p_wzs_T->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    T_wzs_T = parseNextToken(p_wzs_T);
    if (T_wzs_T == NULL || T_wzs_T->type == TOKEN_SEMICOLON) {
        S_wzs_d->where = SRATable(TableReference_make(name_of_wzs_table, NULL));
        return S_wzs_d;
    }
    if(!matchToken(p_wzs_T, TOKEN_RESERVED_WORD, "where")){
        strcpy(p_wzs_T->parserMessage, "语法错误.");
        return NULL;
    }
    Expression *Where_expr = parseExpressionRD(p_wzs_T);
    if (p_wzs_T->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    T_wzs_T = parseNextToken(p_wzs_T);
    if (T_wzs_T == NULL || T_wzs_T->type == TOKEN_SEMICOLON) {
        TableReference_t *wzs_rrr = TableReference_make(name_of_wzs_table, NULL);
        SRA_t *wzs_t =  SRATable(wzs_rrr);
        SRA_t *wzs_s = SRASelect(wzs_t, Where_expr);
        S_wzs_d->where = wzs_s;
        return S_wzs_d;
    }
};
