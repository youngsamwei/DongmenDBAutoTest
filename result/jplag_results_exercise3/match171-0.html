#include <parser/statement.h>

sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){

    char *tttableName = NULL;
    //delete
    TokenT *tttoken = parseNextToken(parser);
    if(!matchToken(parser,TOKEN_RESERVED_WORD,"delete")){
        return NULL;
    }

    //table Name
    tttoken = parseNextToken(parser);
    if(tttoken->type == TOKEN_WORD){
        tttableName = new_id_name();
        strcpy(tttableName,tttoken->text);
    }else{
        return NULL;
    }

    //where
    tttoken = parseEatAndNextToken(parser);
    if(!matchToken(parser,TOKEN_RESERVED_WORD,"where")){
        return NULL;
    }

    //
    TableReference_t *ttref1 = TableReference_make(tttableName,NULL);
    SRA_t *ttwhere = SRATable(ttref1);

    //where����
    tttoken = parseNextToken(parser);
    Expression *ttwhereExpr = parseExpressionRD(parser);
    ttwhere = SRASelect(ttwhere,ttwhereExpr);
    if(parser->parserStateType ==PARSER_WRONG){
        return NULL;
    }

    sql_stmt_delete *ttsqlStmtDelete = (sql_stmt_delete*)calloc(sizeof(sql_stmt_delete),1);
    ttsqlStmtDelete->tableName = tttableName;
    SRA_print(ttwhere);
    ttsqlStmtDelete->where = ttwhere;

    return ttsqlStmtDelete;
};
