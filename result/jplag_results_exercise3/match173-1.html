#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *UPDATE <table_name> SET <field1> = <expr1>[, <field2 = <expr2>, ..]
 * WHERE <logical_expr>
 * 解析获得 sql_stmt_delete 结构
 */


sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){
    char *tableName = NULL;
    arraylist *fields = arraylist_create(); //set fields 被更新的字段列表
    arraylist *fieldsExpr = arraylist_create();  //set fields expression 新值(表达式)列表

    if (!matchToken(parser, TOKEN_RESERVED_WORD, "delete")) {
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }
    TokenT *token = parseNextToken(parser);//首先判断当前解析器中的词是否为 NULL,如果是，就获取下一个 token;否则就返回当前词
     //判断下一个TOken是不是from
     token =parseNextToken(parser);
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "from"))
    {
        strcpy(parser->parserMessage,"invalid sql: missing field that you want to update.");
        return NULL;
    }
     //保存下表名，如果不是表名，提示错误信息
     token =parseNextToken(parser);
   if(token->type==TOKEN_WORD)
   {
       tableName=new_id_name();
       strcpy(tableName,token->text);

   }
    else {
       strcpy(parser->parserMessage,"invalid sql: missing table name");
       return  NULL;
   }
    SRA_t *tablesExpr = parseTablesExpr(parser);


    //匹配 where，matchToken(parser,TOKEN_RESERVED_WORD,"where")
    // ，注意不是所有update语句都伴随一个where表达式
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "where")){
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }

    //使用parseExpressionRD(parse)获取值（或者表达式），
    // 并包裹在SRA_t中（使用SRA_SELECT构造），作为sql_stmt_update->where的值。
    // （语句若不伴随where表达式则需要以一个SRA_TABLE代替，
    // 具体实现可参考 exp_01_stmt_parser/exp_01_03_select.c）

    printf("1111\n");
    Expression *whereExpr = parseExpressionRD(parser);
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    printf("222\n");
    //解析获得 sql_stmt_delete 结构
    //创建sql_stmt_update指针，开空间，对各字段进行赋值，
    // 返回sql_stmt_update结构体
    sql_stmt_delete *sqlStmtDelete = (sql_stmt_delete *)calloc(sizeof(sql_stmt_delete),1);
    sqlStmtDelete->tableName = tableName; //保存上面update语句的student
    printf("33\n");
    SRA_t *select = SRASelect(tablesExpr, whereExpr);SRA_t *where;  //保存上面update语句的 sage = 22
    printf("55\n");
    sqlStmtDelete->where = select;printf("44\n");
    return sqlStmtDelete;


};
