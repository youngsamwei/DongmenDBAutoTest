#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */

sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser) {
    char *whichTable = NULL;//表的名字

    TokenT *token = parseNextToken(parser);//判断delete
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "delete")){
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }

    token = parseNextToken(parser);//析别表的名字
    if(token->type = TOKEN_WORD){
        whichTable = new_id_name();
        strcpy(whichTable,token->text);//将表的名字赋值给whichtable
        token = parseEatAndNextToken(parser);
    }else {
        strcpy(parser->parserMessage,"invalid sql: missing table name.");
        return  NULL;
    }
    //别名为空
    TableReference_t *TableReference_make2 = TableReference_make(whichTable,NULL);
    SRA_t *whereSelect = SRATable(TableReference_make2);

    token = parseNextToken(parser);//处理where和之后的语句
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "where")){
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }
    token = parseNextToken(parser);
    Expression *WhereExpression  = parseExpressionRD(parser);
    whereSelect = SRASelect(whereSelect, WhereExpression);//把一整串语句放入到whereSelect中去保存
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }


    sql_stmt_delete *Delete = (sql_stmt_delete *)calloc(sizeof(sql_stmt_delete),1);
    Delete->tableName = whichTable;
    SRA_print(whereSelect);//物理架构
   Delete->where = whereSelect;

    return Delete;
};
