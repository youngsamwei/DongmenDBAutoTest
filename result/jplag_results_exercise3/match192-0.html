#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */

sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){
     char *tablename;
    arraylist *fields=arraylist_create();
    arraylist *fieldsExpr=arraylist_create();
    SRA_t *findword=NULL;

    TokenT *token = parseNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "delete")){
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }
    //if(!matchToken(parser, TOKEN_RESERVED_WORD, "from")){
      //  strcpy(parser->parserMessage, "语法错误.缺少from.");
        //return NULL;
    //}
    token=parseNextToken(parser);
    if(token->type==TOKEN_WORD)
    {
        tablename=new_id_name();
        strcpy(tablename,token->text);
    }
    token=parseEatAndNextToken(parser);
    Expression* wheredelete;
    if(matchToken(parser,TOKEN_RESERVED_WORD,"where"))
    {
        wheredelete=parseExpressionRD(parser);
    }



    TableReference_t *ref=TableReference_make(tablename,NULL);
    findword=SRATable(ref);

    if(wheredelete!=NULL){
        findword=SRASelect(findword,wheredelete);
    }
    sql_stmt_delete* sql_delete=(sql_stmt_delete*)malloc(sizeof(sql_stmt_delete));

    sql_delete->tableName=tablename;
    sql_delete->where=findword;
    return sql_delete;
};
