#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */

sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){
    char *tn;


    SRA_t *wh=NULL;

    TokenT *tok = parseNextToken(parser);
    if(!matchToken(parser, TOKEN_RESERVED_WORD, "delete")){
        strcpy(parser->parserMessage, "语法错误.");
        return NULL;
    }


    tok=parseNextToken(parser);
    if(tok->type==TOKEN_WORD)
    {
        tn=new_id_name();
        strcpy(tn,tok->text);
    }

    tok=parseEatAndNextToken(parser);

    Expression* wc;
    if(matchToken(parser,TOKEN_RESERVED_WORD,"where")){
        wc=parseExpressionRD(parser);
    }
    TableReference_t *tf=TableReference_make(tn,NULL);
    wh=SRATable(tf);

    if(wc!=NULL){
        wh=SRASelect(wh,wc);
    }
    sql_stmt_delete* sd=(sql_stmt_delete*)malloc(sizeof(sql_stmt_delete));

    sd->tableName=tn;
    sd->where=wh;
    return sd;

};
