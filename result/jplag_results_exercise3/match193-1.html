#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */

sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){
    char *table_name = NULL;

    if(!matchToken(parser,TOKEN_RESERVED_WORD,"delete")){ //匹配 delete 关键词
        return NULL;
    }
    TokenT *token = parseNextToken(parser);
    if(token->type == TOKEN_WORD){ //获取表名
        table_name = new_id_name();
        strcpy(table_name,token->text);
    } else{
        strcpy(parser->parserMessage,"无效的SQL: 缺失表名.");
        return NULL;
    }
    sql_stmt_delete *sql_Stmt_Delete = (sql_stmt_delete *)malloc(sizeof(sql_stmt_delete));
    sql_Stmt_Delete->tableName = table_name;
    TableReference_t *ref =   TableReference_make(table_name, NULL);
    SRA_t *tables_Expr =  SRATable(ref);
    token = parseEatAndNextToken(parser);
    if (token == NULL || token->type == TOKEN_SEMICOLON) { //说明语句已结束
        sql_Stmt_Delete->where = tables_Expr;//该条语句无where条件,直接将sqlStmtUpdate->where赋值为tablesExpr即可
        return sql_Stmt_Delete;
    }

    if(!matchToken(parser,TOKEN_RESERVED_WORD,"where")){ //匹配 where 关键词
        strcpy(parser->parserMessage,"无效的SQL: 缺失where.");
        return NULL;
    }
    token = parseNextToken(parser);
    Expression *where_Expr = parseExpressionRD(parser);
    if (parser->parserStateType == PARSER_WRONG) {
        return NULL;
    }
    SRA_t *where = SRASelect(tables_Expr, where_Expr);
    sql_Stmt_Delete->where = where;
    return sql_Stmt_Delete;
};
