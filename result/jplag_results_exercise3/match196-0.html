#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */

char *parseTableName(ParserT *parser) {
    TokenT *token = parseNextToken(parser);
    if (token->type == TOKEN_WORD) {
        char *tableName = new_id_name();
        strcpy(tableName, token->text);
        parseEatAndNextToken(parser);
        return tableName;
    } else {
        strcpy(parser->parserMessage, "语法错误，请检查输入.");
        return NULL;
    }

}

sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){
    char * TBname = new_id_name();
    sql_stmt_delete* sss_Deltor = (sql_stmt_delete*)malloc(sizeof(sql_stmt_delete));
    TokenT *sss_token = parseNextToken(parser);
    SRA_t *where = NULL;

    if(!matchToken(parser, TOKEN_RESERVED_WORD, "delete")) {
        strcpy(parser->parserMessage, "Not delete!");
        return NULL;
    }
	
    sss_token = parseNextToken(parser);
    if(sss_token->type == TOKEN_WORD) {
        strcpy(TBname, sss_token->text);
        sss_Deltor->tableName = TBname;
    }

    Expression *expr = NULL;
    sss_token = parseEatAndNextToken(parser);
    if(sss_token != NULL && matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
        expr = parseExpressionRD(parser);
    }

    TableReference_t *TBref = TableReference_make(TBname, NULL);
    where = SRATable(TBref);
    if(expr != NULL) {
        where = SRASelect(where, expr);
    }
    sss_Deltor->tableName = TBname;
    sss_Deltor->where = where;
    return sss_Deltor;
};
