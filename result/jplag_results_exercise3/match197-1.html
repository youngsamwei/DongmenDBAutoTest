#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */
//delecct from student where student.sage=19

sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){

    char *tbN_sh = NULL;//tableName
    //delete
    TokenT *tk_key = parseNextToken(parser);
    if(!matchToken(parser,TOKEN_RESERVED_WORD,"delete")){
        strcpy(parser->parserMessage,"error1.");
        return NULL;
    }
    //from
    /*token = parseNextToken(parser);
    if(!matchToken(parser,TOKEN_RESERVED_WORD,"from")){
        strcpy(parser->parserMessage,"error2.");
        return NULL;
    }*/
    //表名
    tk_key = parseNextToken(parser);
    if(tk_key->type == TOKEN_WORD){
        tbN_sh = new_id_name();
        strcpy(tbN_sh,tk_key->text);
    }else{
        strcpy(parser->parserMessage,"error3.");
        return NULL;
    }
    //表名为空
    TableReference_t *ref1_sh = TableReference_make(tbN_sh,NULL);
    SRA_t *where_sh = SRATable(ref1_sh);
    //where
    tk_key = parseEatAndNextToken(parser);
    if(!matchToken(parser,TOKEN_RESERVED_WORD,"where")){
        strcpy(parser->parserMessage,"error4");
        return NULL;
    }
    //where条件

    tk_key = parseNextToken(parser);
    Expression *whereExpr_sh = parseExpressionRD(parser);
    where_sh = SRASelect(where_sh,whereExpr_sh);
    if(parser->parserStateType ==PARSER_WRONG){
        return NULL;
    }

    sql_stmt_delete *SqlStmtDelete_sh = (sql_stmt_delete*)calloc(sizeof(sql_stmt_delete),1);
    SqlStmtDelete_sh->tableName = tbN_sh;
    SRA_print(where_sh);
    SqlStmtDelete_sh->where = where_sh;

    return SqlStmtDelete_sh;
};
