#include <parser/statement.h>

/**
 * 在现有实现基础上，实现delete from子句
 *
 *  支持的delete语法：
 *
 *  DELETE FROM <table_nbame>
 *  WHERE <logical_expr>
 *
 * 解析获得 sql_stmt_delete 结构
 */

sql_stmt_delete *parse_sql_stmt_delete(ParserT *parser){
    char *cccccssssstableName = NULL;
    SRA_t *cccccssssswhere = NULL;

    TokenT *cccccssssstoken;

    // delete
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "delete")) {
        return NULL;
    }

    // table_name
    cccccssssstoken = parseNextToken(parser);
    if (cccccssssstoken && cccccssssstoken->type == TOKEN_WORD) {
        cccccssssstableName = new_id_name();
        strcpy(cccccssssstableName, cccccssssstoken->text);
    } else {
        strcpy(parser->parserMessage, "invalid sql: missing table name.");
        return NULL;
    }
    parseEatAndNextToken(parser);

    // where <logical_expr>
    if (!matchToken(parser, TOKEN_RESERVED_WORD, "where")) {
        return NULL;
    }
    SRA_t *cccccssssstable = SRATable(TableReference_make(cccccssssstableName, NULL));
    Expression *cccccssssscond = parseExpressionRD(parser);
    cccccssssswhere = SRASelect(cccccssssstable, cccccssssscond);

    // return
    sql_stmt_delete *cccccssssssqlStmtDelete = (sql_stmt_delete *) calloc(sizeof(sql_stmt_delete), 1);
    cccccssssssqlStmtDelete->tableName = cccccssssstableName;
    cccccssssssqlStmtDelete->where = cccccssssswhere;

    return cccccssssssqlStmtDelete;
};
